098    @Override
099    public void fillQueueForKey(String keyName,
100        Queue<EncryptedKeyVersion> keyQueue, int numEKVs) throws IOException {
101      checkNotNull(keyName, "keyName");
102      Map<String, String> params = new HashMap<String, String>();
103      params.put(KMSRESTConstants.EEK_OP, KMSRESTConstants.EEK_GENERATE);
104      params.put(KMSRESTConstants.EEK_NUM_KEYS, "" + numEKVs);
105      URL url = createURL(KMSRESTConstants.KEY_RESOURCE, keyName,
106          KMSRESTConstants.EEK_SUB_RESOURCE, params);
107      HttpURLConnection conn = createConnection(url, HTTP_GET);
108      conn.setRequestProperty(CONTENT_TYPE, APPLICATION_JSON_MIME);
109      List response = call(conn, null,
110          HttpURLConnection.HTTP_OK, List.class);
111      List<EncryptedKeyVersion> ekvs =
112          parseJSONEncKeyVersion(keyName, response);
113      keyQueue.addAll(ekvs);
114    }
115  }

125  @SuppressWarnings("rawtypes")
126  private static List<EncryptedKeyVersion>
127      parseJSONEncKeyVersion(String keyName, List valueList) {
128    List<EncryptedKeyVersion> ekvs = new LinkedList<EncryptedKeyVersion>();
129    if (!valueList.isEmpty()) {
130      for (Object values : valueList) {
131        Map valueMap = (Map) values;
132
133        String versionName = checkNotNull(
134                (String) valueMap.get(KMSRESTConstants.VERSION_NAME_FIELD),
135                KMSRESTConstants.VERSION_NAME_FIELD);
136
137        byte[] iv = Base64.decodeBase64(checkNotNull(
138                (String) valueMap.get(KMSRESTConstants.IV_FIELD),
139                KMSRESTConstants.IV_FIELD));
140
141        Map encValueMap = checkNotNull((Map)
142                valueMap.get(KMSRESTConstants.ENCRYPTED_KEY_VERSION_FIELD),
143                KMSRESTConstants.ENCRYPTED_KEY_VERSION_FIELD);
144
145        String encVersionName = checkNotNull((String)
146                encValueMap.get(KMSRESTConstants.VERSION_NAME_FIELD),
147                KMSRESTConstants.VERSION_NAME_FIELD);
148
149        byte[] encKeyMaterial = Base64.decodeBase64(checkNotNull((String)
150                encValueMap.get(KMSRESTConstants.MATERIAL_FIELD),
151                KMSRESTConstants.MATERIAL_FIELD));
152
153        ekvs.add(new KMSEncryptedKeyVersion(keyName, versionName, iv,
154            encVersionName, encKeyMaterial));
155      }
156    }
157    return ekvs;
158  }

160  private static KeyVersion parseJSONKeyVersion(Map valueMap) {
161    KeyVersion keyVersion = null;
162    if (!valueMap.isEmpty()) {
163      byte[] material = (valueMap.containsKey(KMSRESTConstants.MATERIAL_FIELD))
164          ? Base64.decodeBase64((String) valueMap.get(KMSRESTConstants.MATERIAL_FIELD))
165          : null;
166      String versionName = (String)valueMap.get(KMSRESTConstants.VERSION_NAME_FIELD);
167      String keyName = (String)valueMap.get(KMSRESTConstants.NAME_FIELD);
168      keyVersion = new KMSKeyVersion(keyName, versionName, material);
169    }
170    return keyVersion;
171  }

243  private static class TimeoutConnConfigurator
244          implements ConnectionConfigurator {
245    private ConnectionConfigurator cc;
246    private int timeout;

248    /**
249     * Sets the timeout and wraps another connection configurator
250     * @param timeout - will set both connect and read timeouts - in seconds
251     * @param cc - another configurator to wrap - may be null
252     */
253    public TimeoutConnConfigurator(int timeout, ConnectionConfigurator cc) {
254      this.timeout = timeout;
255      this.cc = cc;
256    }
257
258    /**
259     * Calls the wrapped configure() method, then sets timeouts
260     * @param conn the {@link HttpURLConnection} instance to configure.
261     * @return the connection
262     * @throws IOException
263     */
264    @Override
265    public HttpURLConnection configure(HttpURLConnection conn)
266            throws IOException {
267      if (cc != null) {
268        conn = cc.configure(conn);
269      }
270      conn.setConnectTimeout(timeout * 1000);  // conversion to milliseconds
271      conn.setReadTimeout(timeout * 1000);
272      return conn;
273    }

276  public KMSClientProvider(URI uri, Configuration conf) throws IOException {
277    Path path = ProviderUtils.unnestUri(uri);
278    URL url = path.toUri().toURL();
279    kmsUrl = createServiceURL(url);
280    if ("https".equalsIgnoreCase(url.getProtocol())) {
281      sslFactory = new SSLFactory(SSLFactory.Mode.CLIENT, conf);
282      try {
283        sslFactory.init();
284      } catch (GeneralSecurityException ex) {
285        throw new IOException(ex);
286      }
287    }
288    int timeout = conf.getInt(TIMEOUT_ATTR, DEFAULT_TIMEOUT);
289    configurator = new TimeoutConnConfigurator(timeout, sslFactory);
290    encKeyVersionQueue =
291        new ValueQueue<KeyProviderCryptoExtension.EncryptedKeyVersion>(
292            conf.getInt(
293                CommonConfigurationKeysPublic.KMS_CLIENT_ENC_KEY_CACHE_SIZE,
294                CommonConfigurationKeysPublic.
295                    KMS_CLIENT_ENC_KEY_CACHE_SIZE_DEFAULT),
296            conf.getFloat(
297                CommonConfigurationKeysPublic.
298                    KMS_CLIENT_ENC_KEY_CACHE_LOW_WATERMARK,
299                CommonConfigurationKeysPublic.
300                    KMS_CLIENT_ENC_KEY_CACHE_LOW_WATERMARK_DEFAULT),
301            conf.getInt(
302                CommonConfigurationKeysPublic.
303                    KMS_CLIENT_ENC_KEY_CACHE_EXPIRY_MS,
304                CommonConfigurationKeysPublic.
305                    KMS_CLIENT_ENC_KEY_CACHE_EXPIRY_DEFAULT),
306            conf.getInt(
307                CommonConfigurationKeysPublic.
308                    KMS_CLIENT_ENC_KEY_CACHE_NUM_REFILL_THREADS,
309                CommonConfigurationKeysPublic.
310                    KMS_CLIENT_ENC_KEY_CACHE_NUM_REFILL_THREADS_DEFAULT),
311            new EncryptedQueueRefiller());
312  }

314  private String createServiceURL(URL url) throws IOException {
315    String str = url.toExternalForm();
316    if (str.endsWith("/")) {
317      str = str.substring(0, str.length() - 1);
318    }
319    return new URL(str + KMSRESTConstants.SERVICE_VERSION + "/").
320        toExternalForm();
321  }

323  private URL createURL(String collection, String resource, String subResource,
324      Map<String, ?> parameters) throws IOException {
325    try {
326      StringBuilder sb = new StringBuilder();
327      sb.append(kmsUrl);
328      sb.append(collection);
329      if (resource != null) {
330        sb.append("/").append(URLEncoder.encode(resource, UTF8));
331      }
332      if (subResource != null) {
333        sb.append("/").append(subResource);
334      }
335      URIBuilder uriBuilder = new URIBuilder(sb.toString());
336      if (parameters != null) {
337        for (Map.Entry<String, ?> param : parameters.entrySet()) {
338          Object value = param.getValue();
339          if (value instanceof String) {
340            uriBuilder.addParameter(param.getKey(), (String) value);
341          } else {
342            for (String s : (String[]) value) {
343              uriBuilder.addParameter(param.getKey(), s);
344            }
345          }
346        }
347      }
348      return uriBuilder.build().toURL();
349    } catch (URISyntaxException ex) {
350      throw new IOException(ex);
351    }
352  }

354  private HttpURLConnection configureConnection(HttpURLConnection conn)
355      throws IOException {
356    if (sslFactory != null) {
357      HttpsURLConnection httpsConn = (HttpsURLConnection) conn;
358      try {
359        httpsConn.setSSLSocketFactory(sslFactory.createSSLSocketFactory());
360      } catch (GeneralSecurityException ex) {
361        throw new IOException(ex);
362      }
363      httpsConn.setHostnameVerifier(sslFactory.getHostnameVerifier());
364    }
365    return conn;
366  }

368  private HttpURLConnection createConnection(URL url, String method)
369      throws IOException {
370    HttpURLConnection conn;
371    try {
372      AuthenticatedURL authUrl = new AuthenticatedURL(new PseudoAuthenticator(),
373          configurator);
374      conn = authUrl.openConnection(url, new AuthenticatedURL.Token());
375    } catch (AuthenticationException ex) {
376      throw new IOException(ex);
377    }
378    conn.setUseCaches(false);
379    conn.setRequestMethod(method);
380    if (method.equals(HTTP_POST) || method.equals(HTTP_PUT)) {
381      conn.setDoOutput(true);
382    }
383    conn = configureConnection(conn);
384    return conn;
385  }

399  @SuppressWarnings("unchecked")
400  private static void validateResponse(HttpURLConnection conn, int expected)
401      throws IOException {
402    int status = conn.getResponseCode();
403    if (status != expected) {
404      InputStream es = null;
405      try {
406        es = conn.getErrorStream();
407        ObjectMapper mapper = new ObjectMapper();
408        Map json = mapper.readValue(es, Map.class);
409        String exClass = (String) json.get(
410            KMSRESTConstants.ERROR_EXCEPTION_JSON);
411        String exMsg = (String)
412            json.get(KMSRESTConstants.ERROR_MESSAGE_JSON);
413        Exception toThrow;
414        try {
415          ClassLoader cl = KMSClientProvider.class.getClassLoader();
416          Class klass = cl.loadClass(exClass);
417          Constructor constr = klass.getConstructor(String.class);
418          toThrow = (Exception) constr.newInstance(exMsg);
419        } catch (Exception ex) {
420          toThrow = new IOException(MessageFormat.format(
421              "HTTP status [{0}], {1}", status, conn.getResponseMessage()));
422        }
423        throwEx(toThrow);
424      } finally {
425        if (es != null) {
426          es.close();
427        }
428      }
429    }
430  }

432  private static <T> T call(HttpURLConnection conn, Map jsonOutput,
433      int expectedResponse, Class<T> klass)
434      throws IOException {
435    T ret = null;
436    try {
437      if (jsonOutput != null) {
438        writeJson(jsonOutput, conn.getOutputStream());
439      }
440    } catch (IOException ex) {
441      conn.getInputStream().close();
442      throw ex;
443    }
444    validateResponse(conn, expectedResponse);
445    if (APPLICATION_JSON_MIME.equalsIgnoreCase(conn.getContentType())
446        && klass != null) {
447      ObjectMapper mapper = new ObjectMapper();
448      InputStream is = null;
449      try {
450        is = conn.getInputStream();
451        ret = mapper.readValue(is, klass);
452      } catch (IOException ex) {
453        if (is != null) {
454          is.close();
455        }
456        throw ex;
457      } finally {
458        if (is != null) {
459          is.close();
460        }
461      }
462    }
463    return ret;
464  }

509  // breaking keyNames into sets to keep resulting URL undler 2000 chars
510  private List<String[]> createKeySets(String[] keyNames) {
511    List<String[]> list = new ArrayList<String[]>();
512    List<String> batch = new ArrayList<String>();
513    int batchLen = 0;
514    for (String name : keyNames) {
515      int additionalLen = KMSRESTConstants.KEY_OP.length() + 1 + name.length();
516      batchLen += additionalLen;
517      // topping at 1500 to account for initial URL and encoded names
518      if (batchLen > 1500) {
519        list.add(batch.toArray(new String[batch.size()]));
520        batch = new ArrayList<String>();
521        batchLen = additionalLen;
522      }
523      batch.add(name);
524    }
525    if (!batch.isEmpty()) {
526      list.add(batch.toArray(new String[batch.size()]));
527    }
528    return list;
529  }

533  public Metadata[] getKeysMetadata(String ... keyNames) throws IOException {
534    List<Metadata> keysMetadata = new ArrayList<Metadata>();
535    List<String[]> keySets = createKeySets(keyNames);
536    for (String[] keySet : keySets) {
537      if (keyNames.length > 0) {
538        Map<String, Object> queryStr = new HashMap<String, Object>();
539        queryStr.put(KMSRESTConstants.KEY_OP, keySet);
540        URL url = createURL(KMSRESTConstants.KEYS_METADATA_RESOURCE, null,
541            null, queryStr);
542        HttpURLConnection conn = createConnection(url, HTTP_GET);
543        List<Map> list = call(conn, null, HttpURLConnection.HTTP_OK, List.class);
544        for (Map map : list) {
545          keysMetadata.add(parseJSONMetadata(map));
546        }
547      }
548    }
549    return keysMetadata.toArray(new Metadata[keysMetadata.size()]);
550  }

552  private KeyVersion createKeyInternal(String name, byte[] material,
553      Options options)
554      throws NoSuchAlgorithmException, IOException {
555    checkNotEmpty(name, "name");
556    checkNotNull(options, "options");
557    Map<String, Object> jsonKey = new HashMap<String, Object>();
558    jsonKey.put(KMSRESTConstants.NAME_FIELD, name);
559    jsonKey.put(KMSRESTConstants.CIPHER_FIELD, options.getCipher());
560    jsonKey.put(KMSRESTConstants.LENGTH_FIELD, options.getBitLength());
561    if (material != null) {
562      jsonKey.put(KMSRESTConstants.MATERIAL_FIELD,
563          Base64.encodeBase64String(material));
564    }
565    if (options.getDescription() != null) {
566      jsonKey.put(KMSRESTConstants.DESCRIPTION_FIELD,
567          options.getDescription());
568    }
569    if (options.getAttributes() != null && !options.getAttributes().isEmpty()) {
570      jsonKey.put(KMSRESTConstants.ATTRIBUTES_FIELD, options.getAttributes());
571    }
572    URL url = createURL(KMSRESTConstants.KEYS_RESOURCE, null, null, null);
573    HttpURLConnection conn = createConnection(url, HTTP_POST);
574    conn.setRequestProperty(CONTENT_TYPE, APPLICATION_JSON_MIME);
575    Map response = call(conn, jsonKey, HttpURLConnection.HTTP_CREATED,
576        Map.class);
577    return parseJSONKeyVersion(response);
578  }

644  @SuppressWarnings("rawtypes")
645  @Override
646  public KeyVersion decryptEncryptedKey(
647      EncryptedKeyVersion encryptedKeyVersion) throws IOException,
648                                                      GeneralSecurityException {
649    checkNotNull(encryptedKeyVersion.getEncryptionKeyVersionName(),
650        "versionName");
651    checkNotNull(encryptedKeyVersion.getEncryptedKeyIv(), "iv");
652    Preconditions.checkArgument(
653        encryptedKeyVersion.getEncryptedKeyVersion().getVersionName()
654            .equals(KeyProviderCryptoExtension.EEK),
655        "encryptedKey version name must be '%s', is '%s'",
656        KeyProviderCryptoExtension.EEK,
657        encryptedKeyVersion.getEncryptedKeyVersion().getVersionName()
658    );
659    checkNotNull(encryptedKeyVersion.getEncryptedKeyVersion(), "encryptedKey");
660    Map<String, String> params = new HashMap<String, String>();
661    params.put(KMSRESTConstants.EEK_OP, KMSRESTConstants.EEK_DECRYPT);
662    Map<String, Object> jsonPayload = new HashMap<String, Object>();
663    jsonPayload.put(KMSRESTConstants.NAME_FIELD,
664        encryptedKeyVersion.getEncryptionKeyName());
665    jsonPayload.put(KMSRESTConstants.IV_FIELD, Base64.encodeBase64String(
666        encryptedKeyVersion.getEncryptedKeyIv()));
667    jsonPayload.put(KMSRESTConstants.MATERIAL_FIELD, Base64.encodeBase64String(
668            encryptedKeyVersion.getEncryptedKeyVersion().getMaterial()));
669    URL url = createURL(KMSRESTConstants.KEY_VERSION_RESOURCE,
670        encryptedKeyVersion.getEncryptionKeyVersionName(),
671        KMSRESTConstants.EEK_SUB_RESOURCE, params);
672    HttpURLConnection conn = createConnection(url, HTTP_POST);
673    conn.setRequestProperty(CONTENT_TYPE, APPLICATION_JSON_MIME);
674    Map response =
675        call(conn, jsonPayload, HttpURLConnection.HTTP_OK, Map.class);
676    return parseJSONKeyVersion(response);
677  }

680  public List<KeyVersion> getKeyVersions(String name) throws IOException {
681    checkNotEmpty(name, "name");
682    URL url = createURL(KMSRESTConstants.KEY_RESOURCE, name,
683        KMSRESTConstants.VERSIONS_SUB_RESOURCE, null);
684    HttpURLConnection conn = createConnection(url, HTTP_GET);
685    List response = call(conn, null, HttpURLConnection.HTTP_OK, List.class);
686    List<KeyVersion> versions = null;
687    if (!response.isEmpty()) {
688      versions = new ArrayList<KeyVersion>();
689      for (Object obj : response) {
690        versions.add(parseJSONKeyVersion((Map) obj));
691      }
692    }
693    return versions;
694  }
