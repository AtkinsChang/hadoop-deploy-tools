022import java.security.GeneralSecurityException;
023import java.security.SecureRandom;
024
025import javax.crypto.Cipher;

033/**
034 * A KeyProvider with Cryptographic Extensions specifically for generating
035 * and decrypting encrypted encryption keys.
036 * 
037 */
038@InterfaceAudience.Private
039public class KeyProviderCryptoExtension extends
040    KeyProviderExtension<KeyProviderCryptoExtension.CryptoExtension> {
041
042  /**
043   * Designates an encrypted encryption key, or EEK.
044   */
045  public static final String EEK = "EEK";
046  /**
047   * Designates a decrypted encrypted encryption key, that is, an encryption key
048   * (EK).
049   */
050  public static final String EK = "EK";
051
052  /**
053   * An encrypted encryption key (EEK) and related information. An EEK must be
054   * decrypted using the key's encryption key before it can be used.
055   */
056  public static class EncryptedKeyVersion {
057    private String encryptionKeyName;
058    private String encryptionKeyVersionName;
059    private byte[] encryptedKeyIv;
060    private KeyVersion encryptedKeyVersion;
061
062    /**
063     * Create a new EncryptedKeyVersion.
064     *
065     * @param keyName                  Name of the encryption key used to
066     *                                 encrypt the encrypted key.
067     * @param encryptionKeyVersionName Version name of the encryption key used
068     *                                 to encrypt the encrypted key.
069     * @param encryptedKeyIv           Initialization vector of the encrypted
070     *                                 key. The IV of the encryption key used to
071     *                                 encrypt the encrypted key is derived from
072     *                                 this IV.
073     * @param encryptedKeyVersion      The encrypted encryption key version.
074     */
075    protected EncryptedKeyVersion(String keyName,
076        String encryptionKeyVersionName, byte[] encryptedKeyIv,
077        KeyVersion encryptedKeyVersion) {
078      this.encryptionKeyName = keyName;
079      this.encryptionKeyVersionName = encryptionKeyVersionName;
080      this.encryptedKeyIv = encryptedKeyIv;
081      this.encryptedKeyVersion = encryptedKeyVersion;
082    }
083
084    /**
085     * Factory method to create a new EncryptedKeyVersion that can then be
086     * passed into {@link #decryptEncryptedKey}. Note that the fields of the
087     * returned EncryptedKeyVersion will only partially be populated; it is not
088     * necessarily suitable for operations besides decryption.
089     *
090     * @param encryptionKeyVersionName Version name of the encryption key used
091     *                                 to encrypt the encrypted key.
092     * @param encryptedKeyIv           Initialization vector of the encrypted
093     *                                 key. The IV of the encryption key used to
094     *                                 encrypt the encrypted key is derived from
095     *                                 this IV.
096     * @param encryptedKeyMaterial     Key material of the encrypted key.
097     * @return EncryptedKeyVersion suitable for decryption.
098     */
099    public static EncryptedKeyVersion createForDecryption(String
100        encryptionKeyVersionName, byte[] encryptedKeyIv,
101        byte[] encryptedKeyMaterial) {
102      KeyVersion encryptedKeyVersion = new KeyVersion(null, EEK,
103          encryptedKeyMaterial);
104      return new EncryptedKeyVersion(null, encryptionKeyVersionName,
105          encryptedKeyIv, encryptedKeyVersion);
106    }
107
108    /**
109     * @return Name of the encryption key used to encrypt the encrypted key.
110     */
111    public String getEncryptionKeyName() {
112      return encryptionKeyName;
113    }
114
115    /**
116     * @return Version name of the encryption key used to encrypt the encrypted
117     * key.
118     */
119    public String getEncryptionKeyVersionName() {
120      return encryptionKeyVersionName;
121    }
122
123    /**
124     * @return Initialization vector of the encrypted key. The IV of the
125     * encryption key used to encrypt the encrypted key is derived from this
126     * IV.
127     */
128    public byte[] getEncryptedKeyIv() {
129      return encryptedKeyIv;
130    }
131
132    /**
133     * @return The encrypted encryption key version.
134     */
135    public KeyVersion getEncryptedKeyVersion() {
136      return encryptedKeyVersion;
137    }
138
139    /**
140     * Derive the initialization vector (IV) for the encryption key from the IV
141     * of the encrypted key. This derived IV is used with the encryption key to
142     * decrypt the encrypted key.
143     * <p/>
144     * The alternative to this is using the same IV for both the encryption key
145     * and the encrypted key. Even a simple symmetric transformation like this
146     * improves security by avoiding IV re-use. IVs will also be fairly unique
147     * among different EEKs.
148     *
149     * @param encryptedKeyIV of the encrypted key (i.e. {@link
150     * #getEncryptedKeyIv()})
151     * @return IV for the encryption key
152     */
153    protected static byte[] deriveIV(byte[] encryptedKeyIV) {
154      byte[] rIv = new byte[encryptedKeyIV.length];
155      // Do a simple XOR transformation to flip all the bits
156      for (int i = 0; i < encryptedKeyIV.length; i++) {
157        rIv[i] = (byte) (encryptedKeyIV[i] ^ 0xff);
158      }
159      return rIv;
160    }
161  }
162
163  /**
164   * CryptoExtension is a type of Extension that exposes methods to generate
165   * EncryptedKeys and to decrypt the same.
166   */
167  public interface CryptoExtension extends KeyProviderExtension.Extension {
168
169    /**
170     * Calls to this method allows the underlying KeyProvider to warm-up any
171     * implementation specific caches used to store the Encrypted Keys.
172     * @param keyNames Array of Key Names
173     */
174    public void warmUpEncryptedKeys(String... keyNames)
175        throws IOException;
176
177    /**
178     * Generates a key material and encrypts it using the given key version name
179     * and initialization vector. The generated key material is of the same
180     * length as the <code>KeyVersion</code> material of the latest key version
181     * of the key and is encrypted using the same cipher.
182     * <p/>
183     * NOTE: The generated key is not stored by the <code>KeyProvider</code>
184     * 
185     * @param encryptionKeyName
186     *          The latest KeyVersion of this key's material will be encrypted.
187     * @return EncryptedKeyVersion with the generated key material, the version
188     *         name is 'EEK' (for Encrypted Encryption Key)
189     * @throws IOException
190     *           thrown if the key material could not be generated
191     * @throws GeneralSecurityException
192     *           thrown if the key material could not be encrypted because of a
193     *           cryptographic issue.
194     */
195    public EncryptedKeyVersion generateEncryptedKey(
196        String encryptionKeyName) throws IOException,
197        GeneralSecurityException;
198
199    /**
200     * Decrypts an encrypted byte[] key material using the given a key version
201     * name and initialization vector.
202     * 
203     * @param encryptedKeyVersion
204     *          contains keyVersionName and IV to decrypt the encrypted key
205     *          material
206     * @return a KeyVersion with the decrypted key material, the version name is
207     *         'EK' (For Encryption Key)
208     * @throws IOException
209     *           thrown if the key material could not be decrypted
210     * @throws GeneralSecurityException
211     *           thrown if the key material could not be decrypted because of a
212     *           cryptographic issue.
213     */
214    public KeyVersion decryptEncryptedKey(
215        EncryptedKeyVersion encryptedKeyVersion) throws IOException,
216        GeneralSecurityException;
217  }
218
219  private static class DefaultCryptoExtension implements CryptoExtension {
220
221    private final KeyProvider keyProvider;
222
223    private DefaultCryptoExtension(KeyProvider keyProvider) {
224      this.keyProvider = keyProvider;
225    }
226
227    @Override
228    public EncryptedKeyVersion generateEncryptedKey(String encryptionKeyName)
229        throws IOException, GeneralSecurityException {
230      // Fetch the encryption key
231      KeyVersion encryptionKey = keyProvider.getCurrentKey(encryptionKeyName);
232      Preconditions.checkNotNull(encryptionKey,
233          "No KeyVersion exists for key '%s' ", encryptionKeyName);
234      // Generate random bytes for new key and IV
235      Cipher cipher = Cipher.getInstance("AES/CTR/NoPadding");
236      SecureRandom random = SecureRandom.getInstance("SHA1PRNG");
237      final byte[] newKey = new byte[encryptionKey.getMaterial().length];
238      random.nextBytes(newKey);
239      final byte[] iv = random.generateSeed(cipher.getBlockSize());
240      // Encryption key IV is derived from new key's IV
241      final byte[] encryptionIV = EncryptedKeyVersion.deriveIV(iv);
242      // Encrypt the new key
243      cipher.init(Cipher.ENCRYPT_MODE,
244          new SecretKeySpec(encryptionKey.getMaterial(), "AES"),
245          new IvParameterSpec(encryptionIV));
246      final byte[] encryptedKey = cipher.doFinal(newKey);
247      return new EncryptedKeyVersion(encryptionKeyName,
248          encryptionKey.getVersionName(), iv,
249          new KeyVersion(encryptionKey.getName(), EEK, encryptedKey));
250    }
251
252    @Override
253    public KeyVersion decryptEncryptedKey(
254        EncryptedKeyVersion encryptedKeyVersion) throws IOException,
255        GeneralSecurityException {
256      // Fetch the encryption key material
257      final String encryptionKeyVersionName =
258          encryptedKeyVersion.getEncryptionKeyVersionName();
259      final KeyVersion encryptionKey =
260          keyProvider.getKeyVersion(encryptionKeyVersionName);
261      Preconditions.checkNotNull(encryptionKey,
262          "KeyVersion name '%s' does not exist", encryptionKeyVersionName);
263      Preconditions.checkArgument(
264              encryptedKeyVersion.getEncryptedKeyVersion().getVersionName()
265                    .equals(KeyProviderCryptoExtension.EEK),
266                "encryptedKey version name must be '%s', is '%s'",
267                KeyProviderCryptoExtension.EEK,
268                encryptedKeyVersion.getEncryptedKeyVersion().getVersionName()
269            );
270      final byte[] encryptionKeyMaterial = encryptionKey.getMaterial();
271      // Encryption key IV is determined from encrypted key's IV
272      final byte[] encryptionIV =
273          EncryptedKeyVersion.deriveIV(encryptedKeyVersion.getEncryptedKeyIv());
274      // Init the cipher with encryption key parameters
275      Cipher cipher = Cipher.getInstance("AES/CTR/NoPadding");
276      cipher.init(Cipher.DECRYPT_MODE,
277          new SecretKeySpec(encryptionKeyMaterial, "AES"),
278          new IvParameterSpec(encryptionIV));
279      // Decrypt the encrypted key
280      final KeyVersion encryptedKV =
281          encryptedKeyVersion.getEncryptedKeyVersion();
282      final byte[] decryptedKey = cipher.doFinal(encryptedKV.getMaterial());
283      return new KeyVersion(encryptionKey.getName(), EK, decryptedKey);
284    }
285
286    @Override
287    public void warmUpEncryptedKeys(String... keyNames)
288        throws IOException {
289      // NO-OP since the default version does not cache any keys
290    }
291
292  }
293
294  /**
295   * This constructor is to be used by sub classes that provide
296   * delegating/proxying functionality to the {@link KeyProviderCryptoExtension}
297   * @param keyProvider
298   * @param extension
299   */
300  protected KeyProviderCryptoExtension(KeyProvider keyProvider,
301      CryptoExtension extension) {
302    super(keyProvider, extension);
303  }
304
305  /**
306   * Notifies the Underlying CryptoExtension implementation to warm up any
307   * implementation specific caches for the specified KeyVersions
308   * @param keyNames Arrays of key Names
309   */
310  public void warmUpEncryptedKeys(String... keyNames)
311      throws IOException {
312    getExtension().warmUpEncryptedKeys(keyNames);
313  }
314
315  /**
316   * Generates a key material and encrypts it using the given key version name
317   * and initialization vector. The generated key material is of the same
318   * length as the <code>KeyVersion</code> material and is encrypted using the
319   * same cipher.
320   * <p/>
321   * NOTE: The generated key is not stored by the <code>KeyProvider</code>
322   *
323   * @param encryptionKeyName The latest KeyVersion of this key's material will
324   * be encrypted.
325   * @return EncryptedKeyVersion with the generated key material, the version
326   * name is 'EEK' (for Encrypted Encryption Key)
327   * @throws IOException thrown if the key material could not be generated
328   * @throws GeneralSecurityException thrown if the key material could not be 
329   * encrypted because of a cryptographic issue.
330   */
331  public EncryptedKeyVersion generateEncryptedKey(String encryptionKeyName)
332      throws IOException,
333                                           GeneralSecurityException {
334    return getExtension().generateEncryptedKey(encryptionKeyName);
335  }
336
337  /**
338   * Decrypts an encrypted byte[] key material using the given a key version
339   * name and initialization vector.
340   *
341   * @param encryptedKey contains keyVersionName and IV to decrypt the encrypted 
342   * key material
343   * @return a KeyVersion with the decrypted key material, the version name is
344   * 'EK' (For Encryption Key)
345   * @throws IOException thrown if the key material could not be decrypted
346   * @throws GeneralSecurityException thrown if the key material could not be 
347   * decrypted because of a cryptographic issue.
348   */
349  public KeyVersion decryptEncryptedKey(EncryptedKeyVersion encryptedKey) 
350      throws IOException, GeneralSecurityException {
351    return getExtension().decryptEncryptedKey(encryptedKey);
352  }
353
354  /**
355   * Creates a <code>KeyProviderCryptoExtension</code> using a given 
356   * {@link KeyProvider}.
357   * <p/>
358   * If the given <code>KeyProvider</code> implements the 
359   * {@link CryptoExtension} interface the <code>KeyProvider</code> itself
360   * will provide the extension functionality, otherwise a default extension
361   * implementation will be used.
362   * 
363   * @param keyProvider <code>KeyProvider</code> to use to create the 
364   * <code>KeyProviderCryptoExtension</code> extension.
365   * @return a <code>KeyProviderCryptoExtension</code> instance using the
366   * given <code>KeyProvider</code>.
367   */
368  public static KeyProviderCryptoExtension createKeyProviderCryptoExtension(
369      KeyProvider keyProvider) {
370    CryptoExtension cryptoExtension = (keyProvider instanceof CryptoExtension)
371                         ? (CryptoExtension) keyProvider
372                         : new DefaultCryptoExtension(keyProvider);
373    return new KeyProviderCryptoExtension(keyProvider, cryptoExtension);
374  }
