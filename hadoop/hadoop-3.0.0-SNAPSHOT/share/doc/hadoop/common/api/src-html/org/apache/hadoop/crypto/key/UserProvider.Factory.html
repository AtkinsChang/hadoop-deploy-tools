019package org.apache.hadoop.crypto.key;
020
021import java.io.IOException;
022import java.net.URI;
023import java.util.ArrayList;
024import java.util.Date;
025import java.util.HashMap;
026import java.util.List;
027import java.util.Map;
028
029import org.apache.hadoop.classification.InterfaceAudience;
030import org.apache.hadoop.conf.Configuration;
031import org.apache.hadoop.io.Text;
032import org.apache.hadoop.security.Credentials;
033import org.apache.hadoop.security.UserGroupInformation;
034
035/**
036 * A KeyProvider factory for UGIs. It uses the credentials object associated
037 * with the current user to find keys. This provider is created using a
038 * URI of "user:///".
039 */
040@InterfaceAudience.Private
041public class UserProvider extends KeyProvider {
042  public static final String SCHEME_NAME = "user";
043  private final UserGroupInformation user;
044  private final Credentials credentials;
045  private final Map<String, Metadata> cache = new HashMap<String, Metadata>();
046
047  private UserProvider() throws IOException {
048    user = UserGroupInformation.getCurrentUser();
049    credentials = user.getCredentials();
050  }
051
052  @Override
053  public boolean isTransient() {
054    return true;
055  }
056
057  @Override
058  public synchronized KeyVersion getKeyVersion(String versionName) {
059    byte[] bytes = credentials.getSecretKey(new Text(versionName));
060    if (bytes == null) {
061      return null;
062    }
063    return new KeyVersion(versionName, bytes);
064  }
065
066  @Override
067  public synchronized Metadata getMetadata(String name) throws IOException {
068    if (cache.containsKey(name)) {
069      return cache.get(name);
070    }
071    byte[] serialized = credentials.getSecretKey(new Text(name));
072    if (serialized == null) {
073      return null;
074    }
075    Metadata result = new Metadata(serialized);
076    cache.put(name, result);
077    return result;
078  }
079
080  @Override
081  public synchronized KeyVersion createKey(String name, byte[] material,
082                               Options options) throws IOException {
083    Text nameT = new Text(name);
084    if (credentials.getSecretKey(nameT) != null) {
085      throw new IOException("Key " + name + " already exists in " + this);
086    }
087    if (options.getBitLength() != 8 * material.length) {
088      throw new IOException("Wrong key length. Required " +
089          options.getBitLength() + ", but got " + (8 * material.length));
090    }
091    Metadata meta = new Metadata(options.getCipher(), options.getBitLength(),
092        options.getDescription(), new Date(), 1);
093    cache.put(name, meta);
094    String versionName = buildVersionName(name, 0);
095    credentials.addSecretKey(nameT, meta.serialize());
096    credentials.addSecretKey(new Text(versionName), material);
097    return new KeyVersion(versionName, material);
098  }
099
100  @Override
101  public synchronized void deleteKey(String name) throws IOException {
102    Metadata meta = getMetadata(name);
103    if (meta == null) {
104      throw new IOException("Key " + name + " does not exist in " + this);
105    }
106    for(int v=0; v < meta.getVersions(); ++v) {
107      credentials.removeSecretKey(new Text(buildVersionName(name, v)));
108    }
109    credentials.removeSecretKey(new Text(name));
110    cache.remove(name);
111  }
112
113  @Override
114  public synchronized KeyVersion rollNewVersion(String name,
115                                    byte[] material) throws IOException {
116    Metadata meta = getMetadata(name);
117    if (meta == null) {
118      throw new IOException("Key " + name + " not found");
119    }
120    if (meta.getBitLength() != 8 * material.length) {
121      throw new IOException("Wrong key length. Required " +
122          meta.getBitLength() + ", but got " + (8 * material.length));
123    }
124    int nextVersion = meta.addVersion();
125    credentials.addSecretKey(new Text(name), meta.serialize());
126    String versionName = buildVersionName(name, nextVersion);
127    credentials.addSecretKey(new Text(versionName), material);
128    return new KeyVersion(versionName, material);
129  }
130
131  @Override
132  public String toString() {
133    return SCHEME_NAME + ":///";
134  }
135
136  @Override
137  public synchronized void flush() {
138    user.addCredentials(credentials);
139  }
140
141  public static class Factory extends KeyProviderFactory {
142
143    @Override
144    public KeyProvider createProvider(URI providerName,
145                                      Configuration conf) throws IOException {
146      if (SCHEME_NAME.equals(providerName.getScheme())) {
147        return new UserProvider();
148      }
149      return null;
150    }
151  }
152
153  @Override
154  public synchronized List<String> getKeys() throws IOException {
155    List<String> list = new ArrayList<String>();
156    List<Text> keys = credentials.getAllSecretKeys();
157    for (Text key : keys) {
158      if (key.find("@") == -1) {
159        list.add(key.toString());
160      }
161    }
162    return list;
163  }
164
165  @Override
166  public synchronized List<KeyVersion> getKeyVersions(String name) throws IOException {
167      List<KeyVersion> list = new ArrayList<KeyVersion>();
168      Metadata km = getMetadata(name);
169      if (km != null) {
170        int latestVersion = km.getVersions();
171        for (int i = 0; i < latestVersion; i++) {
172          KeyVersion v = getKeyVersion(buildVersionName(name, i));
173          if (v != null) {
174            list.add(v);
175          }
176        }
177      }
178      return list;
179  }
180}
