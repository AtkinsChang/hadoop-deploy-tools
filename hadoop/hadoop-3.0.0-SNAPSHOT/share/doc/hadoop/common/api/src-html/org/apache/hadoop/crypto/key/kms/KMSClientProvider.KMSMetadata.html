073  private static KeyVersion parseJSONKeyVersion(Map valueMap) {
074    KeyVersion keyVersion = null;
075    if (!valueMap.isEmpty()) {
076      byte[] material = (valueMap.containsKey(KMSRESTConstants.MATERIAL_FIELD))
077          ? Base64.decodeBase64((String) valueMap.get(KMSRESTConstants.MATERIAL_FIELD))
078          : null;
079      keyVersion = new KMSKeyVersion((String)
080          valueMap.get(KMSRESTConstants.VERSION_NAME_FIELD), material);
081    }
082    return keyVersion;
083  }
084
085  private static Metadata parseJSONMetadata(Map valueMap) {
086    Metadata metadata = null;
087    if (!valueMap.isEmpty()) {
088      metadata = new KMSMetadata(
089          (String) valueMap.get(KMSRESTConstants.CIPHER_FIELD),
090          (Integer) valueMap.get(KMSRESTConstants.LENGTH_FIELD),
091          (String) valueMap.get(KMSRESTConstants.DESCRIPTION_FIELD),
092          new Date((Long) valueMap.get(KMSRESTConstants.CREATED_FIELD)),
093          (Integer) valueMap.get(KMSRESTConstants.VERSIONS_FIELD));
094    }
095    return metadata;
096  }

149  public KMSClientProvider(URI uri, Configuration conf) throws IOException {
150    Path path = unnestUri(uri);
151    URL url = path.toUri().toURL();
152    kmsUrl = createServiceURL(url);
153    if ("https".equalsIgnoreCase(url.getProtocol())) {
154      sslFactory = new SSLFactory(SSLFactory.Mode.CLIENT, conf);
155      try {
156        sslFactory.init();
157      } catch (GeneralSecurityException ex) {
158        throw new IOException(ex);
159      }
160    }
161  }
162
163  private String createServiceURL(URL url) throws IOException {
164    String str = url.toExternalForm();
165    if (str.endsWith("/")) {
166      str = str.substring(0, str.length() - 1);
167    }
168    return new URL(str + KMSRESTConstants.SERVICE_VERSION + "/").
169        toExternalForm();
170  }
171
172  private URL createURL(String collection, String resource, String subResource,
173      Map<String, ?> parameters) throws IOException {
174    try {
175      StringBuilder sb = new StringBuilder();
176      sb.append(kmsUrl);
177      sb.append(collection);
178      if (resource != null) {
179        sb.append("/").append(URLEncoder.encode(resource, UTF8));
180      }
181      if (subResource != null) {
182        sb.append("/").append(subResource);
183      }
184      URIBuilder uriBuilder = new URIBuilder(sb.toString());
185      if (parameters != null) {
186        for (Map.Entry<String, ?> param : parameters.entrySet()) {
187          Object value = param.getValue();
188          if (value instanceof String) {
189            uriBuilder.addParameter(param.getKey(), (String) value);
190          } else {
191            for (String s : (String[]) value) {
192              uriBuilder.addParameter(param.getKey(), s);
193            }
194          }
195        }
196      }
197      return uriBuilder.build().toURL();
198    } catch (URISyntaxException ex) {
199      throw new IOException(ex);
200    }
201  }

203  private HttpURLConnection configureConnection(HttpURLConnection conn)
204      throws IOException {
205    if (sslFactory != null) {
206      HttpsURLConnection httpsConn = (HttpsURLConnection) conn;
207      try {
208        httpsConn.setSSLSocketFactory(sslFactory.createSSLSocketFactory());
209      } catch (GeneralSecurityException ex) {
210        throw new IOException(ex);
211      }
212      httpsConn.setHostnameVerifier(sslFactory.getHostnameVerifier());
213    }
214    return conn;
215  }
216
217  private HttpURLConnection createConnection(URL url, String method)
218      throws IOException {
219    HttpURLConnection conn;
220    try {
221      AuthenticatedURL authUrl = new AuthenticatedURL(new PseudoAuthenticator(),
222          sslFactory);
223      conn = authUrl.openConnection(url, new AuthenticatedURL.Token());
224    } catch (AuthenticationException ex) {
225      throw new IOException(ex);
226    }
227    conn.setUseCaches(false);
228    conn.setRequestMethod(method);
229    if (method.equals(HTTP_POST) || method.equals(HTTP_PUT)) {
230      conn.setDoOutput(true);
231    }
232    conn = configureConnection(conn);
233    return conn;
234  }

248  @SuppressWarnings("unchecked")
249  private static void validateResponse(HttpURLConnection conn, int expected)
250      throws IOException {
251    int status = conn.getResponseCode();
252    if (status != expected) {
253      InputStream es = null;
254      try {
255        es = conn.getErrorStream();
256        ObjectMapper mapper = new ObjectMapper();
257        Map json = mapper.readValue(es, Map.class);
258        String exClass = (String) json.get(
259            KMSRESTConstants.ERROR_EXCEPTION_JSON);
260        String exMsg = (String)
261            json.get(KMSRESTConstants.ERROR_MESSAGE_JSON);
262        Exception toThrow;
263        try {
264          ClassLoader cl = KMSClientProvider.class.getClassLoader();
265          Class klass = cl.loadClass(exClass);
266          Constructor constr = klass.getConstructor(String.class);
267          toThrow = (Exception) constr.newInstance(exMsg);
268        } catch (Exception ex) {
269          toThrow = new IOException(MessageFormat.format(
270              "HTTP status [{0}], {1}", status, conn.getResponseMessage()));
271        }
272        throwEx(toThrow);
273      } finally {
274        if (es != null) {
275          es.close();
276        }
277      }
278    }
279  }
280
281  private static <T> T call(HttpURLConnection conn, Map jsonOutput,
282      int expectedResponse, Class<T> klass)
283      throws IOException {
284    T ret = null;
285    try {
286      if (jsonOutput != null) {
287        writeJson(jsonOutput, conn.getOutputStream());
288      }
289    } catch (IOException ex) {
290      conn.getInputStream().close();
291      throw ex;
292    }
293    validateResponse(conn, expectedResponse);
294    if (APPLICATION_JSON_MIME.equalsIgnoreCase(conn.getContentType())
295        && klass != null) {
296      ObjectMapper mapper = new ObjectMapper();
297      InputStream is = null;
298      try {
299        is = conn.getInputStream();
300        ret = mapper.readValue(is, klass);
301      } catch (IOException ex) {
302        if (is != null) {
303          is.close();
304        }
305        throw ex;
306      } finally {
307        if (is != null) {
308          is.close();
309        }
310      }
311    }
312    return ret;
313  }

358  // breaking keyNames into sets to keep resulting URL undler 2000 chars
359  private List<String[]> createKeySets(String[] keyNames) {
360    List<String[]> list = new ArrayList<String[]>();
361    List<String> batch = new ArrayList<String>();
362    int batchLen = 0;
363    for (String name : keyNames) {
364      int additionalLen = KMSRESTConstants.KEY_OP.length() + 1 + name.length();
365      batchLen += additionalLen;
366      // topping at 1500 to account for initial URL and encoded names
367      if (batchLen > 1500) {
368        list.add(batch.toArray(new String[batch.size()]));
369        batch = new ArrayList<String>();
370        batchLen = additionalLen;
371      }
372      batch.add(name);
373    }
374    if (!batch.isEmpty()) {
375      list.add(batch.toArray(new String[batch.size()]));
376    }
377    return list;
378  }
379
380  @Override
381  @SuppressWarnings("unchecked")
382  public Metadata[] getKeysMetadata(String ... keyNames) throws IOException {
383    List<Metadata> keysMetadata = new ArrayList<Metadata>();
384    List<String[]> keySets = createKeySets(keyNames);
385    for (String[] keySet : keySets) {
386      if (keyNames.length > 0) {
387        Map<String, Object> queryStr = new HashMap<String, Object>();
388        queryStr.put(KMSRESTConstants.KEY_OP, keySet);
389        URL url = createURL(KMSRESTConstants.KEYS_METADATA_RESOURCE, null,
390            null, queryStr);
391        HttpURLConnection conn = createConnection(url, HTTP_GET);
392        List<Map> list = call(conn, null, HttpURLConnection.HTTP_OK, List.class);
393        for (Map map : list) {
394          keysMetadata.add(parseJSONMetadata(map));
395        }
396      }
397    }
398    return keysMetadata.toArray(new Metadata[keysMetadata.size()]);
399  }

401  private KeyVersion createKeyInternal(String name, byte[] material,
402      Options options)
403      throws NoSuchAlgorithmException, IOException {
404    checkNotEmpty(name, "name");
405    checkNotNull(options, "options");
406    Map<String, Object> jsonKey = new HashMap<String, Object>();
407    jsonKey.put(KMSRESTConstants.NAME_FIELD, name);
408    jsonKey.put(KMSRESTConstants.CIPHER_FIELD, options.getCipher());
409    jsonKey.put(KMSRESTConstants.LENGTH_FIELD, options.getBitLength());
410    if (material != null) {
411      jsonKey.put(KMSRESTConstants.MATERIAL_FIELD,
412          Base64.encodeBase64String(material));
413    }
414    if (options.getDescription() != null) {
415      jsonKey.put(KMSRESTConstants.DESCRIPTION_FIELD,
416          options.getDescription());
417    }
418    URL url = createURL(KMSRESTConstants.KEYS_RESOURCE, null, null, null);
419    HttpURLConnection conn = createConnection(url, HTTP_POST);
420    conn.setRequestProperty(CONTENT_TYPE, APPLICATION_JSON_MIME);
421    Map response = call(conn, jsonKey, HttpURLConnection.HTTP_CREATED,
422        Map.class);
423    return parseJSONKeyVersion(response);
424  }

443  private KeyVersion rollNewVersionInternal(String name, byte[] material)
444      throws NoSuchAlgorithmException, IOException {
445    checkNotEmpty(name, "name");
446    Map<String, String> jsonMaterial = new HashMap<String, String>();
447    if (material != null) {
448      jsonMaterial.put(KMSRESTConstants.MATERIAL_FIELD,
449          Base64.encodeBase64String(material));
450    }
451    URL url = createURL(KMSRESTConstants.KEY_RESOURCE, name, null, null);
452    HttpURLConnection conn = createConnection(url, HTTP_POST);
453    conn.setRequestProperty(CONTENT_TYPE, APPLICATION_JSON_MIME);
454    Map response = call(conn, jsonMaterial,
455        HttpURLConnection.HTTP_OK, Map.class);
456    return parseJSONKeyVersion(response);
457  }

484    List<KeyVersion> versions = null;
485    if (!response.isEmpty()) {
486      versions = new ArrayList<KeyVersion>();

488        versions.add(parseJSONKeyVersion((Map) obj));

490    }
