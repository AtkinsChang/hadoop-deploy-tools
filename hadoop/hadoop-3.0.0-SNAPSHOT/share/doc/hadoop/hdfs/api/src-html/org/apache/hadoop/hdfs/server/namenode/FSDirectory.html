177  FSDirectory(FSImage fsImage, FSNamesystem ns, Configuration conf) {
178    this.dirLock = new ReentrantReadWriteLock(true); // fair
179    this.cond = dirLock.writeLock().newCondition();
180    rootDir = createRoot(ns);
181    inodeMap = INodeMap.newInstance(rootDir);
182    this.fsImage = fsImage;
183    int configuredLimit = conf.getInt(
184        DFSConfigKeys.DFS_LIST_LIMIT, DFSConfigKeys.DFS_LIST_LIMIT_DEFAULT);
185    this.lsLimit = configuredLimit>0 ?
186        configuredLimit : DFSConfigKeys.DFS_LIST_LIMIT_DEFAULT;
187    this.contentCountLimit = conf.getInt(
188        DFSConfigKeys.DFS_CONTENT_SUMMARY_LIMIT_KEY,
189        DFSConfigKeys.DFS_CONTENT_SUMMARY_LIMIT_DEFAULT);
190    
191    // filesystem limits
192    this.maxComponentLength = conf.getInt(
193        DFSConfigKeys.DFS_NAMENODE_MAX_COMPONENT_LENGTH_KEY,
194        DFSConfigKeys.DFS_NAMENODE_MAX_COMPONENT_LENGTH_DEFAULT);
195    this.maxDirItems = conf.getInt(
196        DFSConfigKeys.DFS_NAMENODE_MAX_DIRECTORY_ITEMS_KEY,
197        DFSConfigKeys.DFS_NAMENODE_MAX_DIRECTORY_ITEMS_DEFAULT);
198    this.inodeXAttrsLimit = conf.getInt(
199        DFSConfigKeys.DFS_NAMENODE_MAX_XATTRS_PER_INODE_KEY,
200        DFSConfigKeys.DFS_NAMENODE_MAX_XATTRS_PER_INODE_DEFAULT);
201    Preconditions.checkArgument(this.inodeXAttrsLimit >= 0,
202        "Cannot set a negative limit on the number of xattrs per inode (%s).",
203        DFSConfigKeys.DFS_NAMENODE_MAX_XATTRS_PER_INODE_KEY);
204    // We need a maximum maximum because by default, PB limits message sizes
205    // to 64MB. This means we can only store approximately 6.7 million entries
206    // per directory, but let's use 6.4 million for some safety.
207    final int MAX_DIR_ITEMS = 64 * 100 * 1000;
208    Preconditions.checkArgument(
209        maxDirItems > 0 && maxDirItems <= MAX_DIR_ITEMS, "Cannot set "
210            + DFSConfigKeys.DFS_NAMENODE_MAX_DIRECTORY_ITEMS_KEY
211            + " to a value less than 0 or greater than " + MAX_DIR_ITEMS);
212
213    int threshold = conf.getInt(
214        DFSConfigKeys.DFS_NAMENODE_NAME_CACHE_THRESHOLD_KEY,
215        DFSConfigKeys.DFS_NAMENODE_NAME_CACHE_THRESHOLD_DEFAULT);
216    NameNode.LOG.info("Caching file names occuring more than " + threshold
217        + " times");
218    nameCache = new NameCache<ByteArray>(threshold);
219    namesystem = ns;
220  }

275  /**
276   * Block until the object is ready to be used.
277   */
278  void waitForReady() {
279    if (!ready) {
280      writeLock();
281      try {
282        while (!ready) {
283          try {
284            cond.await(5000, TimeUnit.MILLISECONDS);
285          } catch (InterruptedException ignored) {
286          }
287        }
288      } finally {
289        writeUnlock();
290      }
291    }

305   * Add the given filename to the fs.
306   * @throws FileAlreadyExistsException
307   * @throws QuotaExceededException
308   * @throws UnresolvedLinkException
309   * @throws SnapshotAccessControlException 
310   */
311  INodeFile addFile(String path, PermissionStatus permissions,
312      short replication, long preferredBlockSize, String clientName,
313      String clientMachine, DatanodeDescriptor clientNode)
314    throws FileAlreadyExistsException, QuotaExceededException,
315      UnresolvedLinkException, SnapshotAccessControlException, AclException {
316    waitForReady();
317
318    long modTime = now();
319    INodeFile newNode = new INodeFile(namesystem.allocateNewInodeId(), null,
320        permissions, modTime, modTime, BlockInfo.EMPTY_ARRAY, replication,
321        preferredBlockSize);
322    newNode.toUnderConstruction(clientName, clientMachine, clientNode);
323
324    boolean added = false;
325    writeLock();
326    try {
327      added = addINode(path, newNode);
328    } finally {
329      writeUnlock();
330    }
331    if (!added) {
332      NameNode.stateChangeLog.info("DIR* addFile: failed to add " + path);
333      return null;
334    }
335
336    if(NameNode.stateChangeLog.isDebugEnabled()) {
337      NameNode.stateChangeLog.debug("DIR* addFile: " + path + " is added");
338    }
339    return newNode;
340  }

344                            PermissionStatus permissions,
345                            List<AclEntry> aclEntries,

347                            long modificationTime,
348                            long atime,

357          modificationTime, BlockInfo.EMPTY_ARRAY, replication,
358          preferredBlockSize);

361    } else {
362      newNode = new INodeFile(id, null, permissions, modificationTime, atime,
363          BlockInfo.EMPTY_ARRAY, replication, preferredBlockSize);
364    }
365
366    try {
367      if (addINode(path, newNode)) {
368        if (aclEntries != null) {
369          AclStorage.updateINodeAcl(newNode, aclEntries,
370            Snapshot.CURRENT_STATE_ID);
371        }
372        return newNode;
373      }
374    } catch (IOException e) {
375      if(NameNode.stateChangeLog.isDebugEnabled()) {
376        NameNode.stateChangeLog.debug(
377            "DIR* FSDirectory.unprotectedAddFile: exception when add " + path
378                + " to the file system", e);
379      }
380    }
381    return null;
382  }

384  /**
385   * Add a block to the file. Returns a reference to the added block.
386   */
387  BlockInfo addBlock(String path, INodesInPath inodesInPath, Block block,
388      DatanodeStorageInfo[] targets) throws IOException {
389    waitForReady();
390
391    writeLock();
392    try {
393      final INodeFile fileINode = inodesInPath.getLastINode().asFile();
394      Preconditions.checkState(fileINode.isUnderConstruction());
395
396      // check quota limits and updated space consumed
397      updateCount(inodesInPath, 0, fileINode.getBlockDiskspace(), true);
398
399      // associate new last block for the file
400      BlockInfoUnderConstruction blockInfo =
401        new BlockInfoUnderConstruction(
402            block,
403            fileINode.getFileReplication(),
404            BlockUCState.UNDER_CONSTRUCTION,
405            targets);
406      getBlockManager().addBlockCollection(blockInfo, fileINode);
407      fileINode.addBlock(blockInfo);
408
409      if(NameNode.stateChangeLog.isDebugEnabled()) {
410        NameNode.stateChangeLog.debug("DIR* FSDirectory.addBlock: "
411            + path + " with " + block
412            + " block is added to the in-memory "
413            + "file system");
414      }
415      return blockInfo;
416    } finally {
417      writeUnlock();
418    }
419  }

438  boolean unprotectedRemoveBlock(String path,
439      INodeFile fileNode, Block block) throws IOException {
440    // modify file-> block and blocksMap
441    // fileNode should be under construction
442    boolean removed = fileNode.removeLastBlock(block);
443    if (!removed) {
444      return false;
445    }
446    getBlockManager().removeBlockFromMap(block);
447
448    if(NameNode.stateChangeLog.isDebugEnabled()) {
449      NameNode.stateChangeLog.debug("DIR* FSDirectory.removeBlock: "
450          +path+" with "+block
451          +" block is removed from the file system");
452    }
453
454    // update space consumed
455    final INodesInPath iip = getINodesInPath4Write(path, true);
456    updateCount(iip, 0, -fileNode.getBlockDiskspace(), true);
457    return true;
458  }

507  /**
508   * Change a path name
509   * 
510   * @param src source path
511   * @param dst destination path
512   * @return true if rename succeeds; false otherwise
513   * @throws QuotaExceededException if the operation violates any quota limit
514   * @throws FileAlreadyExistsException if the src is a symlink that points to dst
515   * @throws SnapshotAccessControlException if path is in RO snapshot
516   * @deprecated See {@link #renameTo(String, String, boolean, Rename...)}
517   */
518  @Deprecated
519  boolean unprotectedRenameTo(String src, String dst, long timestamp)
520    throws QuotaExceededException, UnresolvedLinkException, 
521    FileAlreadyExistsException, SnapshotAccessControlException, IOException {
522    assert hasWriteLock();
523    INodesInPath srcIIP = getINodesInPath4Write(src, false);
524    final INode srcInode = srcIIP.getLastINode();
525    
526    // check the validation of the source
527    if (srcInode == null) {
528      NameNode.stateChangeLog.warn("DIR* FSDirectory.unprotectedRenameTo: "
529          + "failed to rename " + src + " to " + dst
530          + " because source does not exist");
531      return false;
532    } 
533    if (srcIIP.getINodes().length == 1) {
534      NameNode.stateChangeLog.warn("DIR* FSDirectory.unprotectedRenameTo: "
535          +"failed to rename "+src+" to "+dst+ " because source is the root");
536      return false;
537    }
538    
539    // srcInode and its subtree cannot contain snapshottable directories with
540    // snapshots
541    List<INodeDirectorySnapshottable> snapshottableDirs = 
542        new ArrayList<INodeDirectorySnapshottable>();
543    checkSnapshot(srcInode, snapshottableDirs);
544    
545    if (isDir(dst)) {
546      dst += Path.SEPARATOR + new Path(src).getName();
547    }
548    
549    // check the validity of the destination
550    if (dst.equals(src)) {
551      return true;
552    }
553    if (srcInode.isSymlink() && 
554        dst.equals(srcInode.asSymlink().getSymlinkString())) {
555      throw new FileAlreadyExistsException(
556          "Cannot rename symlink "+src+" to its target "+dst);
557    }
558    
559    // dst cannot be directory or a file under src
560    if (dst.startsWith(src) && 
561        dst.charAt(src.length()) == Path.SEPARATOR_CHAR) {
562      NameNode.stateChangeLog.warn("DIR* FSDirectory.unprotectedRenameTo: "
563          + "failed to rename " + src + " to " + dst
564          + " because destination starts with src");
565      return false;
566    }
567    
568    byte[][] dstComponents = INode.getPathComponents(dst);
569    INodesInPath dstIIP = getExistingPathINodes(dstComponents);
570    if (dstIIP.isSnapshot()) {
571      throw new SnapshotAccessControlException(
572          "Modification on RO snapshot is disallowed");
573    }
574    if (dstIIP.getLastINode() != null) {
575      NameNode.stateChangeLog.warn("DIR* FSDirectory.unprotectedRenameTo: "
576                                   +"failed to rename "+src+" to "+dst+ 
577                                   " because destination exists");
578      return false;
579    }
580    INode dstParent = dstIIP.getINode(-2);
581    if (dstParent == null) {
582      NameNode.stateChangeLog.warn("DIR* FSDirectory.unprotectedRenameTo: "
583          +"failed to rename "+src+" to "+dst+ 
584          " because destination's parent does not exist");
585      return false;
586    }
587    
588    // Ensure dst has quota to accommodate rename
589    verifyFsLimitsForRename(srcIIP, dstIIP);
590    verifyQuotaForRename(srcIIP.getINodes(), dstIIP.getINodes());
591    
592    boolean added = false;
593    INode srcChild = srcIIP.getLastINode();
594    final byte[] srcChildName = srcChild.getLocalNameBytes();
595    final boolean isSrcInSnapshot = srcChild.isInLatestSnapshot(
596        srcIIP.getLatestSnapshotId());
597    final boolean srcChildIsReference = srcChild.isReference();
598    
599    // Record the snapshot on srcChild. After the rename, before any new 
600    // snapshot is taken on the dst tree, changes will be recorded in the latest
601    // snapshot of the src tree.
602    if (isSrcInSnapshot) {
603      srcChild = srcChild.recordModification(srcIIP.getLatestSnapshotId());
604      srcIIP.setLastINode(srcChild);
605    }
606    
607    // check srcChild for reference
608    final INodeReference.WithCount withCount;
609    Quota.Counts oldSrcCounts = Quota.Counts.newInstance();
610    int srcRefDstSnapshot = srcChildIsReference ? srcChild.asReference()
611        .getDstSnapshotId() : Snapshot.CURRENT_STATE_ID;
612    if (isSrcInSnapshot) {
613      final INodeReference.WithName withName = 
614          srcIIP.getINode(-2).asDirectory().replaceChild4ReferenceWithName(
615              srcChild, srcIIP.getLatestSnapshotId()); 
616      withCount = (INodeReference.WithCount) withName.getReferredINode();
617      srcChild = withName;
618      srcIIP.setLastINode(srcChild);
619      // get the counts before rename
620      withCount.getReferredINode().computeQuotaUsage(oldSrcCounts, true);
621    } else if (srcChildIsReference) {
622      // srcChild is reference but srcChild is not in latest snapshot
623      withCount = (WithCount) srcChild.asReference().getReferredINode();
624    } else {
625      withCount = null;
626    }
627
628    try {
629      // remove src
630      final long removedSrc = removeLastINode(srcIIP);
631      if (removedSrc == -1) {
632        NameNode.stateChangeLog.warn("DIR* FSDirectory.unprotectedRenameTo: "
633            + "failed to rename " + src + " to " + dst
634            + " because the source can not be removed");
635        return false;
636      }
637      
638      if (dstParent.getParent() == null) {
639        // src and dst file/dir are in the same directory, and the dstParent has
640        // been replaced when we removed the src. Refresh the dstIIP and
641        // dstParent.
642        dstIIP = getExistingPathINodes(dstComponents);
643        dstParent = dstIIP.getINode(-2);
644      }
645      
646      // add src to the destination
647      
648      srcChild = srcIIP.getLastINode();
649      final byte[] dstChildName = dstIIP.getLastLocalName();
650      final INode toDst;
651      if (withCount == null) {
652        srcChild.setLocalName(dstChildName);
653        toDst = srcChild;
654      } else {
655        withCount.getReferredINode().setLocalName(dstChildName);
656        int dstSnapshotId = dstIIP.getLatestSnapshotId();
657        toDst = new INodeReference.DstReference(
658            dstParent.asDirectory(), withCount, dstSnapshotId);
659      }
660      
661      added = addLastINodeNoQuotaCheck(dstIIP, toDst);
662      if (added) {
663        if (NameNode.stateChangeLog.isDebugEnabled()) {
664          NameNode.stateChangeLog.debug("DIR* FSDirectory.unprotectedRenameTo: " 
665              + src + " is renamed to " + dst);
666        }
667        // update modification time of dst and the parent of src
668        final INode srcParent = srcIIP.getINode(-2);
669        srcParent.updateModificationTime(timestamp, srcIIP.getLatestSnapshotId());
670        dstParent = dstIIP.getINode(-2); // refresh dstParent
671        dstParent.updateModificationTime(timestamp, dstIIP.getLatestSnapshotId());
672        // update moved leases with new filename
673        getFSNamesystem().unprotectedChangeLease(src, dst);     
674
675        // update the quota usage in src tree
676        if (isSrcInSnapshot) {
677          // get the counts after rename
678          Quota.Counts newSrcCounts = srcChild.computeQuotaUsage(
679              Quota.Counts.newInstance(), false);
680          newSrcCounts.subtract(oldSrcCounts);
681          srcParent.addSpaceConsumed(newSrcCounts.get(Quota.NAMESPACE),
682              newSrcCounts.get(Quota.DISKSPACE), false);
683        }
684        
685        return true;
686      }
687    } finally {
688      if (!added) {
689        final INodeDirectory srcParent = srcIIP.getINode(-2).asDirectory();
690        final INode oldSrcChild = srcChild;
691        // put it back
692        if (withCount == null) {
693          srcChild.setLocalName(srcChildName);
694        } else if (!srcChildIsReference) { // src must be in snapshot
695          // the withCount node will no longer be used thus no need to update
696          // its reference number here
697          srcChild = withCount.getReferredINode();
698          srcChild.setLocalName(srcChildName);
699        } else {
700          withCount.removeReference(oldSrcChild.asReference());
701          srcChild = new INodeReference.DstReference(
702              srcParent, withCount, srcRefDstSnapshot);
703          withCount.getReferredINode().setLocalName(srcChildName);
704        }
705        
706        if (isSrcInSnapshot) {
707          // srcParent must have snapshot feature since isSrcInSnapshot is true
708          // and src node has been removed from srcParent 
709          srcParent.undoRename4ScrParent(oldSrcChild.asReference(), srcChild);
710        } else {
711          // original srcChild is not in latest snapshot, we only need to add
712          // the srcChild back
713          addLastINodeNoQuotaCheck(srcIIP, srcChild);
714        }
715      }
716    }
717    NameNode.stateChangeLog.warn("DIR* FSDirectory.unprotectedRenameTo: "
718        +"failed to rename "+src+" to "+dst);
719    return false;
720  }

722  /**
723   * Rename src to dst.
724   * See {@link DistributedFileSystem#rename(Path, Path, Options.Rename...)}
725   * for details related to rename semantics and exceptions.
726   * 
727   * @param src source path
728   * @param dst destination path
729   * @param timestamp modification time
730   * @param options Rename options
731   */
732  boolean unprotectedRenameTo(String src, String dst, long timestamp,
733      Options.Rename... options) throws FileAlreadyExistsException,
734      FileNotFoundException, ParentNotDirectoryException,
735      QuotaExceededException, UnresolvedLinkException, IOException {
736    assert hasWriteLock();
737    boolean overwrite = false;
738    if (null != options) {
739      for (Rename option : options) {
740        if (option == Rename.OVERWRITE) {
741          overwrite = true;
742        }
743      }
744    }
745    final String error;
746    final INodesInPath srcIIP = getINodesInPath4Write(src, false);
747    final INode srcInode = srcIIP.getLastINode();
748    // validate source
749    if (srcInode == null) {
750      error = "rename source " + src + " is not found.";
751      NameNode.stateChangeLog.warn("DIR* FSDirectory.unprotectedRenameTo: "
752          + error);
753      throw new FileNotFoundException(error);
754    }
755    if (srcIIP.getINodes().length == 1) {
756      error = "rename source cannot be the root";
757      NameNode.stateChangeLog.warn("DIR* FSDirectory.unprotectedRenameTo: "
758          + error);
759      throw new IOException(error);
760    }
761    // srcInode and its subtree cannot contain snapshottable directories with
762    // snapshots
763    checkSnapshot(srcInode, null);
764    
765    // validate the destination
766    if (dst.equals(src)) {
767      throw new FileAlreadyExistsException(
768          "The source "+src+" and destination "+dst+" are the same");
769    }
770    if (srcInode.isSymlink() && 
771        dst.equals(srcInode.asSymlink().getSymlinkString())) {
772      throw new FileAlreadyExistsException(
773          "Cannot rename symlink "+src+" to its target "+dst);
774    }
775    // dst cannot be a directory or a file under src
776    if (dst.startsWith(src) && 
777        dst.charAt(src.length()) == Path.SEPARATOR_CHAR) {
778      error = "Rename destination " + dst
779          + " is a directory or file under source " + src;
780      NameNode.stateChangeLog.warn("DIR* FSDirectory.unprotectedRenameTo: "
781          + error);
782      throw new IOException(error);
783    }
784    INodesInPath dstIIP = getINodesInPath4Write(dst, false);
785    if (dstIIP.getINodes().length == 1) {
786      error = "rename destination cannot be the root";
787      NameNode.stateChangeLog.warn("DIR* FSDirectory.unprotectedRenameTo: "
788          + error);
789      throw new IOException(error);
790    }
791
792    final INode dstInode = dstIIP.getLastINode();
793    List<INodeDirectorySnapshottable> snapshottableDirs = 
794        new ArrayList<INodeDirectorySnapshottable>();
795    if (dstInode != null) { // Destination exists
796      // It's OK to rename a file to a symlink and vice versa
797      if (dstInode.isDirectory() != srcInode.isDirectory()) {
798        error = "Source " + src + " and destination " + dst
799            + " must both be directories";
800        NameNode.stateChangeLog.warn("DIR* FSDirectory.unprotectedRenameTo: "
801            + error);
802        throw new IOException(error);
803      }
804      if (!overwrite) { // If destination exists, overwrite flag must be true
805        error = "rename destination " + dst + " already exists";
806        NameNode.stateChangeLog.warn("DIR* FSDirectory.unprotectedRenameTo: "
807            + error);
808        throw new FileAlreadyExistsException(error);
809      }
810      if (dstInode.isDirectory()) {
811        final ReadOnlyList<INode> children = dstInode.asDirectory()
812            .getChildrenList(Snapshot.CURRENT_STATE_ID);
813        if (!children.isEmpty()) {
814          error = "rename destination directory is not empty: " + dst;
815          NameNode.stateChangeLog.warn(
816              "DIR* FSDirectory.unprotectedRenameTo: " + error);
817          throw new IOException(error);
818        }
819      }
820      checkSnapshot(dstInode, snapshottableDirs);
821    }
822
823    INode dstParent = dstIIP.getINode(-2);
824    if (dstParent == null) {
825      error = "rename destination parent " + dst + " not found.";
826      NameNode.stateChangeLog.warn("DIR* FSDirectory.unprotectedRenameTo: "
827          + error);
828      throw new FileNotFoundException(error);
829    }
830    if (!dstParent.isDirectory()) {
831      error = "rename destination parent " + dst + " is a file.";
832      NameNode.stateChangeLog.warn("DIR* FSDirectory.unprotectedRenameTo: "
833          + error);
834      throw new ParentNotDirectoryException(error);
835    }
836
837    // Ensure dst has quota to accommodate rename
838    verifyFsLimitsForRename(srcIIP, dstIIP);
839    verifyQuotaForRename(srcIIP.getINodes(), dstIIP.getINodes());
840
841    INode srcChild = srcIIP.getLastINode();
842    final byte[] srcChildName = srcChild.getLocalNameBytes();
843    final boolean isSrcInSnapshot = srcChild.isInLatestSnapshot(
844        srcIIP.getLatestSnapshotId());
845    final boolean srcChildIsReference = srcChild.isReference();
846    
847    // Record the snapshot on srcChild. After the rename, before any new 
848    // snapshot is taken on the dst tree, changes will be recorded in the latest
849    // snapshot of the src tree.
850    if (isSrcInSnapshot) {
851      srcChild = srcChild.recordModification(srcIIP.getLatestSnapshotId());
852      srcIIP.setLastINode(srcChild);
853    }
854    
855    // check srcChild for reference
856    final INodeReference.WithCount withCount;
857    int srcRefDstSnapshot = srcChildIsReference ? srcChild.asReference()
858        .getDstSnapshotId() : Snapshot.CURRENT_STATE_ID;
859    Quota.Counts oldSrcCounts = Quota.Counts.newInstance();    
860    if (isSrcInSnapshot) {
861      final INodeReference.WithName withName = srcIIP.getINode(-2).asDirectory()
862          .replaceChild4ReferenceWithName(srcChild, srcIIP.getLatestSnapshotId()); 
863      withCount = (INodeReference.WithCount) withName.getReferredINode();
864      srcChild = withName;
865      srcIIP.setLastINode(srcChild);
866      // get the counts before rename
867      withCount.getReferredINode().computeQuotaUsage(oldSrcCounts, true);
868    } else if (srcChildIsReference) {
869      // srcChild is reference but srcChild is not in latest snapshot
870      withCount = (WithCount) srcChild.asReference().getReferredINode();
871    } else {
872      withCount = null;
873    }
874    
875    boolean undoRemoveSrc = true;
876    final long removedSrc = removeLastINode(srcIIP);
877    if (removedSrc == -1) {
878      error = "Failed to rename " + src + " to " + dst
879          + " because the source can not be removed";
880      NameNode.stateChangeLog.warn("DIR* FSDirectory.unprotectedRenameTo: "
881          + error);
882      throw new IOException(error);
883    }
884    
885    if (dstParent.getParent() == null) {
886      // src and dst file/dir are in the same directory, and the dstParent has
887      // been replaced when we removed the src. Refresh the dstIIP and
888      // dstParent.
889      dstIIP = getINodesInPath4Write(dst, false);
890    }
891    
892    boolean undoRemoveDst = false;
893    INode removedDst = null;
894    try {
895      if (dstInode != null) { // dst exists remove it
896        if (removeLastINode(dstIIP) != -1) {
897          removedDst = dstIIP.getLastINode();
898          undoRemoveDst = true;
899        }
900      }
901      
902      srcChild = srcIIP.getLastINode();
903
904      final byte[] dstChildName = dstIIP.getLastLocalName();
905      final INode toDst;
906      if (withCount == null) {
907        srcChild.setLocalName(dstChildName);
908        toDst = srcChild;
909      } else {
910        withCount.getReferredINode().setLocalName(dstChildName);
911        int dstSnapshotId = dstIIP.getLatestSnapshotId();
912        toDst = new INodeReference.DstReference(
913            dstIIP.getINode(-2).asDirectory(), withCount, dstSnapshotId);
914      }
915
916      // add src as dst to complete rename
917      if (addLastINodeNoQuotaCheck(dstIIP, toDst)) {
918        undoRemoveSrc = false;
919        if (NameNode.stateChangeLog.isDebugEnabled()) {
920          NameNode.stateChangeLog.debug(
921              "DIR* FSDirectory.unprotectedRenameTo: " + src
922              + " is renamed to " + dst);
923        }
924
925        final INode srcParent = srcIIP.getINode(-2);
926        srcParent.updateModificationTime(timestamp, srcIIP.getLatestSnapshotId());
927        dstParent = dstIIP.getINode(-2);
928        dstParent.updateModificationTime(timestamp, dstIIP.getLatestSnapshotId());
929        // update moved lease with new filename
930        getFSNamesystem().unprotectedChangeLease(src, dst);
931
932        // Collect the blocks and remove the lease for previous dst
933        long filesDeleted = -1;
934        if (removedDst != null) {
935          undoRemoveDst = false;
936          BlocksMapUpdateInfo collectedBlocks = new BlocksMapUpdateInfo();
937          List<INode> removedINodes = new ChunkedArrayList<INode>();
938          filesDeleted = removedDst.cleanSubtree(Snapshot.CURRENT_STATE_ID,
939              dstIIP.getLatestSnapshotId(), collectedBlocks, removedINodes, true)
940              .get(Quota.NAMESPACE);
941          getFSNamesystem().removePathAndBlocks(src, collectedBlocks,
942              removedINodes);
943        }
944
945        if (snapshottableDirs.size() > 0) {
946          // There are snapshottable directories (without snapshots) to be
947          // deleted. Need to update the SnapshotManager.
948          namesystem.removeSnapshottableDirs(snapshottableDirs);
949        }
950        
951        // update the quota usage in src tree
952        if (isSrcInSnapshot) {
953          // get the counts after rename
954          Quota.Counts newSrcCounts = srcChild.computeQuotaUsage(
955              Quota.Counts.newInstance(), false);
956          newSrcCounts.subtract(oldSrcCounts);
957          srcParent.addSpaceConsumed(newSrcCounts.get(Quota.NAMESPACE),
958              newSrcCounts.get(Quota.DISKSPACE), false);
959        }
960        
961        return filesDeleted >= 0;
962      }
963    } finally {
964      if (undoRemoveSrc) {
965        // Rename failed - restore src
966        final INodeDirectory srcParent = srcIIP.getINode(-2).asDirectory();
967        final INode oldSrcChild = srcChild;
968        // put it back
969        if (withCount == null) {
970          srcChild.setLocalName(srcChildName);
971        } else if (!srcChildIsReference) { // src must be in snapshot
972          // the withCount node will no longer be used thus no need to update
973          // its reference number here
974          srcChild = withCount.getReferredINode();
975          srcChild.setLocalName(srcChildName);
976        } else {
977          withCount.removeReference(oldSrcChild.asReference());
978          srcChild = new INodeReference.DstReference(
979              srcParent, withCount, srcRefDstSnapshot);
980          withCount.getReferredINode().setLocalName(srcChildName);
981        }
982        
983        if (srcParent.isWithSnapshot()) {
984          srcParent.undoRename4ScrParent(oldSrcChild.asReference(), srcChild);
985        } else {
986          // srcParent is not an INodeDirectoryWithSnapshot, we only need to add
987          // the srcChild back
988          addLastINodeNoQuotaCheck(srcIIP, srcChild);
989        }
990      }
991      if (undoRemoveDst) {
992        // Rename failed - restore dst
993        if (dstParent.isDirectory() && dstParent.asDirectory().isWithSnapshot()) {
994          dstParent.asDirectory().undoRename4DstParent(removedDst,
995              dstIIP.getLatestSnapshotId());
996        } else {
997          addLastINodeNoQuotaCheck(dstIIP, removedDst);
998        }
999        if (removedDst.isReference()) {
1000          final INodeReference removedDstRef = removedDst.asReference();
1001          final INodeReference.WithCount wc = 
1002              (WithCount) removedDstRef.getReferredINode().asReference();
1003          wc.addReference(removedDstRef);
1004        }
1005      }
1006    }
1007    NameNode.stateChangeLog.warn("DIR* FSDirectory.unprotectedRenameTo: "
1008        + "failed to rename " + src + " to " + dst);
1009    throw new IOException("rename from " + src + " to " + dst + " failed.");
1010  }

1034  Block[] unprotectedSetReplication(String src, short replication,
1035      short[] blockRepls) throws QuotaExceededException,
1036      UnresolvedLinkException, SnapshotAccessControlException {
1037    assert hasWriteLock();
1038
1039    final INodesInPath iip = getINodesInPath4Write(src, true);
1040    final INode inode = iip.getLastINode();
1041    if (inode == null || !inode.isFile()) {
1042      return null;
1043    }
1044    INodeFile file = inode.asFile();
1045    final short oldBR = file.getBlockReplication();
1046
1047    // before setFileReplication, check for increasing block replication.
1048    // if replication > oldBR, then newBR == replication.
1049    // if replication < oldBR, we don't know newBR yet. 
1050    if (replication > oldBR) {
1051      long dsDelta = (replication - oldBR)*(file.diskspaceConsumed()/oldBR);
1052      updateCount(iip, 0, dsDelta, true);
1053    }
1054
1055    file = file.setFileReplication(replication, iip.getLatestSnapshotId(),
1056        inodeMap);
1057    
1058    final short newBR = file.getBlockReplication(); 
1059    // check newBR < oldBR case. 
1060    if (newBR < oldBR) {
1061      long dsDelta = (newBR - oldBR)*(file.diskspaceConsumed()/newBR);
1062      updateCount(iip, 0, dsDelta, true);
1063    }
1064
1065    if (blockRepls != null) {
1066      blockRepls[0] = oldBR;
1067      blockRepls[1] = newBR;
1068    }
1069    return file.getBlocks();
1070  }

1155  /**
1156   * Concat all the blocks from srcs to trg and delete the srcs files
1157   * @param target target file to move the blocks to
1158   * @param srcs list of file to move the blocks from
1159   */
1160  void unprotectedConcat(String target, String [] srcs, long timestamp) 
1161      throws UnresolvedLinkException, QuotaExceededException,
1162      SnapshotAccessControlException, SnapshotException {
1163    assert hasWriteLock();
1164    if (NameNode.stateChangeLog.isDebugEnabled()) {
1165      NameNode.stateChangeLog.debug("DIR* FSNamesystem.concat to "+target);
1166    }
1167    // do the move
1168    
1169    final INodesInPath trgIIP = getINodesInPath4Write(target, true);
1170    final INode[] trgINodes = trgIIP.getINodes();
1171    final INodeFile trgInode = trgIIP.getLastINode().asFile();
1172    INodeDirectory trgParent = trgINodes[trgINodes.length-2].asDirectory();
1173    final int trgLatestSnapshot = trgIIP.getLatestSnapshotId();
1174    
1175    final INodeFile [] allSrcInodes = new INodeFile[srcs.length];
1176    for(int i = 0; i < srcs.length; i++) {
1177      final INodesInPath iip = getINodesInPath4Write(srcs[i]);
1178      final int latest = iip.getLatestSnapshotId();
1179      final INode inode = iip.getLastINode();
1180
1181      // check if the file in the latest snapshot
1182      if (inode.isInLatestSnapshot(latest)) {
1183        throw new SnapshotException("Concat: the source file " + srcs[i]
1184            + " is in snapshot " + latest);
1185      }
1186
1187      // check if the file has other references.
1188      if (inode.isReference() && ((INodeReference.WithCount)
1189          inode.asReference().getReferredINode()).getReferenceCount() > 1) {
1190        throw new SnapshotException("Concat: the source file " + srcs[i]
1191            + " is referred by some other reference in some snapshot.");
1192      }
1193
1194      allSrcInodes[i] = inode.asFile();
1195    }
1196    trgInode.concatBlocks(allSrcInodes);
1197    
1198    // since we are in the same dir - we can use same parent to remove files
1199    int count = 0;
1200    for(INodeFile nodeToRemove: allSrcInodes) {
1201      if(nodeToRemove == null) continue;
1202      
1203      nodeToRemove.setBlocks(null);
1204      trgParent.removeChild(nodeToRemove, trgLatestSnapshot);
1205      inodeMap.remove(nodeToRemove);
1206      count++;
1207    }
1208    
1209    // update inodeMap
1210    removeFromInodeMap(Arrays.asList(allSrcInodes));
1211    
1212    trgInode.setModificationTime(timestamp, trgLatestSnapshot);
1213    trgParent.updateModificationTime(timestamp, trgLatestSnapshot);
1214    // update quota on the parent directory ('count' files removed, 0 space)
1215    unprotectedUpdateCount(trgIIP, trgINodes.length-1, -count, 0);
1216  }

1253  private static boolean deleteAllowed(final INodesInPath iip,
1254      final String src) {
1255    final INode[] inodes = iip.getINodes(); 
1256    if (inodes == null || inodes.length == 0
1257        || inodes[inodes.length - 1] == null) {
1258      if(NameNode.stateChangeLog.isDebugEnabled()) {
1259        NameNode.stateChangeLog.debug("DIR* FSDirectory.unprotectedDelete: "
1260            + "failed to remove " + src + " because it does not exist");
1261      }
1262      return false;
1263    } else if (inodes.length == 1) { // src is the root
1264      NameNode.stateChangeLog.warn("DIR* FSDirectory.unprotectedDelete: "
1265          + "failed to remove " + src
1266          + " because the root is not allowed to be deleted");
1267      return false;
1268    }
1269    return true;
1270  }

1325  /**
1326   * Delete a path from the name space
1327   * Update the count at each ancestor directory with quota
1328   * @param iip the inodes resolved from the path
1329   * @param collectedBlocks blocks collected from the deleted path
1330   * @param removedINodes inodes that should be removed from {@link #inodeMap}
1331   * @param mtime the time the inode is removed
1332   * @return the number of inodes deleted; 0 if no inodes are deleted.
1333   */ 
1334  long unprotectedDelete(INodesInPath iip, BlocksMapUpdateInfo collectedBlocks,
1335      List<INode> removedINodes, long mtime) throws QuotaExceededException {
1336    assert hasWriteLock();
1337
1338    // check if target node exists
1339    INode targetNode = iip.getLastINode();
1340    if (targetNode == null) {
1341      return -1;
1342    }
1343
1344    // record modification
1345    final int latestSnapshot = iip.getLatestSnapshotId();
1346    targetNode = targetNode.recordModification(latestSnapshot);
1347    iip.setLastINode(targetNode);
1348
1349    // Remove the node from the namespace
1350    long removed = removeLastINode(iip);
1351    if (removed == -1) {
1352      return -1;
1353    }
1354
1355    // set the parent's modification time
1356    final INodeDirectory parent = targetNode.getParent();
1357    parent.updateModificationTime(mtime, latestSnapshot);
1358    if (removed == 0) {
1359      return 0;
1360    }
1361    
1362    // collect block
1363    if (!targetNode.isInLatestSnapshot(latestSnapshot)) {
1364      targetNode.destroyAndCollectBlocks(collectedBlocks, removedINodes);
1365    } else {
1366      Quota.Counts counts = targetNode.cleanSubtree(Snapshot.CURRENT_STATE_ID,
1367          latestSnapshot, collectedBlocks, removedINodes, true);
1368      parent.addSpaceConsumed(-counts.get(Quota.NAMESPACE),
1369          -counts.get(Quota.DISKSPACE), true);
1370      removed = counts.get(Quota.NAMESPACE);
1371    }
1372    if (NameNode.stateChangeLog.isDebugEnabled()) {
1373      NameNode.stateChangeLog.debug("DIR* FSDirectory.unprotectedDelete: "
1374          + targetNode.getFullPathName() + " is removed");
1375    }
1376    return removed;
1377  }

1410  /**
1411   * Get a partial listing of the indicated directory
1412   *
1413   * We will stop when any of the following conditions is met:
1414   * 1) this.lsLimit files have been added
1415   * 2) needLocation is true AND enough files have been added such
1416   * that at least this.lsLimit block locations are in the response
1417   *
1418   * @param src the directory name
1419   * @param startAfter the name to start listing after
1420   * @param needLocation if block locations are returned
1421   * @return a partial listing starting after startAfter
1422   */
1423  DirectoryListing getListing(String src, byte[] startAfter,
1424      boolean needLocation) throws UnresolvedLinkException, IOException {
1425    String srcs = normalizePath(src);
1426
1427    readLock();
1428    try {
1429      if (srcs.endsWith(HdfsConstants.SEPARATOR_DOT_SNAPSHOT_DIR)) {
1430        return getSnapshotsListing(srcs, startAfter);
1431      }
1432      final INodesInPath inodesInPath = getLastINodeInPath(srcs, true);
1433      final int snapshot = inodesInPath.getPathSnapshotId();
1434      final INode targetNode = inodesInPath.getINode(0);
1435      if (targetNode == null)
1436        return null;
1437      
1438      if (!targetNode.isDirectory()) {
1439        return new DirectoryListing(
1440            new HdfsFileStatus[]{createFileStatus(HdfsFileStatus.EMPTY_NAME,
1441                targetNode, needLocation, snapshot)}, 0);
1442      }
1443
1444      final INodeDirectory dirInode = targetNode.asDirectory();
1445      final ReadOnlyList<INode> contents = dirInode.getChildrenList(snapshot);
1446      int startChild = INodeDirectory.nextChild(contents, startAfter);
1447      int totalNumChildren = contents.size();
1448      int numOfListing = Math.min(totalNumChildren-startChild, this.lsLimit);
1449      int locationBudget = this.lsLimit;
1450      int listingCnt = 0;
1451      HdfsFileStatus listing[] = new HdfsFileStatus[numOfListing];
1452      for (int i=0; i<numOfListing && locationBudget>0; i++) {
1453        INode cur = contents.get(startChild+i);
1454        listing[i] = createFileStatus(cur.getLocalNameBytes(), cur,
1455            needLocation, snapshot);
1456        listingCnt++;
1457        if (needLocation) {
1458            // Once we  hit lsLimit locations, stop.
1459            // This helps to prevent excessively large response payloads.
1460            // Approximate #locations with locatedBlockCount() * repl_factor
1461            LocatedBlocks blks = 
1462                ((HdfsLocatedFileStatus)listing[i]).getBlockLocations();
1463            locationBudget -= (blks == null) ? 0 :
1464               blks.locatedBlockCount() * listing[i].getReplication();
1465        }
1466      }
1467      // truncate return array if necessary
1468      if (listingCnt < numOfListing) {
1469          listing = Arrays.copyOf(listing, listingCnt);
1470      }
1471      return new DirectoryListing(
1472          listing, totalNumChildren-startChild-listingCnt);
1473    } finally {
1474      readUnlock();
1475    }
1476  }

1478  /**
1479   * Get a listing of all the snapshots of a snapshottable directory
1480   */
1481  private DirectoryListing getSnapshotsListing(String src, byte[] startAfter)
1482      throws UnresolvedLinkException, IOException {
1483    Preconditions.checkState(hasReadLock());
1484    Preconditions.checkArgument(
1485        src.endsWith(HdfsConstants.SEPARATOR_DOT_SNAPSHOT_DIR),
1486        "%s does not end with %s", src, HdfsConstants.SEPARATOR_DOT_SNAPSHOT_DIR);
1487    
1488    final String dirPath = normalizePath(src.substring(0,
1489        src.length() - HdfsConstants.DOT_SNAPSHOT_DIR.length()));
1490    
1491    final INode node = this.getINode(dirPath);
1492    final INodeDirectorySnapshottable dirNode = INodeDirectorySnapshottable
1493        .valueOf(node, dirPath);
1494    final ReadOnlyList<Snapshot> snapshots = dirNode.getSnapshotList();
1495    int skipSize = ReadOnlyList.Util.binarySearch(snapshots, startAfter);
1496    skipSize = skipSize < 0 ? -skipSize - 1 : skipSize + 1;
1497    int numOfListing = Math.min(snapshots.size() - skipSize, this.lsLimit);
1498    final HdfsFileStatus listing[] = new HdfsFileStatus[numOfListing];
1499    for (int i = 0; i < numOfListing; i++) {
1500      Root sRoot = snapshots.get(i + skipSize).getRoot();
1501      listing[i] = createFileStatus(sRoot.getLocalNameBytes(), sRoot,
1502          Snapshot.CURRENT_STATE_ID);
1503    }
1504    return new DirectoryListing(
1505        listing, snapshots.size() - skipSize - numOfListing);
1506  }

1762  /**
1763   * @return the relative path of an inode from one of its ancestors,
1764   *         represented by an array of inodes.
1765   */
1766  private static INode[] getRelativePathINodes(INode inode, INode ancestor) {
1767    // calculate the depth of this inode from the ancestor
1768    int depth = 0;
1769    for (INode i = inode; i != null && !i.equals(ancestor); i = i.getParent()) {
1770      depth++;
1771    }
1772    INode[] inodes = new INode[depth];
1773
1774    // fill up the inodes in the path from this inode to root
1775    for (int i = 0; i < depth; i++) {
1776      if (inode == null) {
1777        NameNode.stateChangeLog.warn("Could not get full path."
1778            + " Corresponding file might have deleted already.");
1779        return null;
1780      }
1781      inodes[depth-i-1] = inode;
1782      inode = inode.getParent();
1783    }
1784    return inodes;
1785  }

1856   * @param commonAncestor Last node in inodes array that is a common ancestor
1857   *          for a INode that is being moved from one location to the other.
1858   *          Pass null if a node is not being moved.
1859   * @throws QuotaExceededException if quota limit is exceeded.
1860   */
1861  private static void verifyQuota(INode[] inodes, int pos, long nsDelta,
1862      long dsDelta, INode commonAncestor) throws QuotaExceededException {
1863    if (nsDelta <= 0 && dsDelta <= 0) {
1864      // if quota is being freed or not being consumed
1865      return;
1866    }
1867
1868    // check existing components in the path
1869    for(int i = (pos > inodes.length? inodes.length: pos) - 1; i >= 0; i--) {
1870      if (commonAncestor == inodes[i]) {
1871        // Stop checking for quota when common ancestor is reached
1872        return;
1873      }
1874      final DirectoryWithQuotaFeature q
1875          = inodes[i].asDirectory().getDirectoryWithQuotaFeature();
1876      if (q != null) { // a directory with quota
1877        try {
1878          q.verifyQuota(nsDelta, dsDelta);
1879        } catch (QuotaExceededException e) {
1880          e.setPathName(getFullPathName(inodes, i));
1881          throw e;
1882        }
1883      }
1884    }
1885  }

1888   * Verify quota for rename operation where srcInodes[srcInodes.length-1] moves
1889   * dstInodes[dstInodes.length-1]
1890   * 
1891   * @param src directory from where node is being moved.
1892   * @param dst directory to where node is moved to.
1893   * @throws QuotaExceededException if quota limit is exceeded.
1894   */
1895  private void verifyQuotaForRename(INode[] src, INode[] dst)
1896      throws QuotaExceededException {
1897    if (!ready || skipQuotaCheck) {
1898      // Do not check quota if edits log is still being processed
1899      return;
1900    }
1901    int i = 0;
1902    while(src[i] == dst[i]) { i++; }
1903    // src[i - 1] is the last common ancestor.
1904
1905    final Quota.Counts delta = src[src.length - 1].computeQuotaUsage();
1906    
1907    // Reduce the required quota by dst that is being removed
1908    final int dstIndex = dst.length - 1;
1909    if (dst[dstIndex] != null) {
1910      delta.subtract(dst[dstIndex].computeQuotaUsage());
1911    }
1912    verifyQuota(dst, dstIndex, delta.get(Quota.NAMESPACE),
1913        delta.get(Quota.DISKSPACE), src[i - 1]);
1914  }

2025  /** Add a node child to the inodes at index pos. 
2026   * Its ancestors are stored at [0, pos-1].
2027   * @return false if the child with this name already exists; 
2028   *         otherwise return true;
2029   * @throws QuotaExceededException is thrown if it violates quota limit
2030   */
2031  private boolean addChild(INodesInPath iip, int pos,
2032      INode child, boolean checkQuota) throws QuotaExceededException {
2033    final INode[] inodes = iip.getINodes();
2034    // Disallow creation of /.reserved. This may be created when loading
2035    // editlog/fsimage during upgrade since /.reserved was a valid name in older
2036    // release. This may also be called when a user tries to create a file
2037    // or directory /.reserved.
2038    if (pos == 1 && inodes[0] == rootDir && isReservedName(child)) {
2039      throw new HadoopIllegalArgumentException(
2040          "File name \"" + child.getLocalName() + "\" is reserved and cannot "
2041              + "be created. If this is during upgrade change the name of the "
2042              + "existing file or directory to another name before upgrading "
2043              + "to the new release.");
2044    }
2045    // The filesystem limits are not really quotas, so this check may appear
2046    // odd. It's because a rename operation deletes the src, tries to add
2047    // to the dest, if that fails, re-adds the src from whence it came.
2048    // The rename code disables the quota when it's restoring to the
2049    // original location becase a quota violation would cause the the item
2050    // to go "poof".  The fs limits must be bypassed for the same reason.
2051    if (checkQuota) {
2052      verifyMaxComponentLength(child.getLocalNameBytes(), inodes, pos);
2053      verifyMaxDirItems(inodes, pos);
2054    }
2055    // always verify inode name
2056    verifyINodeName(child.getLocalNameBytes());
2057    
2058    final Quota.Counts counts = child.computeQuotaUsage();
2059    updateCount(iip, pos,
2060        counts.get(Quota.NAMESPACE), counts.get(Quota.DISKSPACE), checkQuota);
2061    boolean isRename = (child.getParent() != null);
2062    final INodeDirectory parent = inodes[pos-1].asDirectory();
2063    boolean added;
2064    try {
2065      added = parent.addChild(child, true, iip.getLatestSnapshotId());
2066    } catch (QuotaExceededException e) {
2067      updateCountNoQuotaCheck(iip, pos,
2068          -counts.get(Quota.NAMESPACE), -counts.get(Quota.DISKSPACE));
2069      throw e;
2070    }
2071    if (!added) {
2072      updateCountNoQuotaCheck(iip, pos,
2073          -counts.get(Quota.NAMESPACE), -counts.get(Quota.DISKSPACE));
2074    } else {
2075      iip.setINode(pos - 1, child.getParent());
2076      if (!isRename) {
2077        AclStorage.copyINodeDefaultAcl(child);
2078      }
2079      addToInodeMap(child);
2080    }
2081    return added;
2082  }

2094   * Remove the last inode in the path from the namespace.
2095   * Count of each ancestor with quota is also updated.
2096   * @return -1 for failing to remove;
2097   *          0 for removing a reference whose referred inode has other 
2098   *            reference nodes;
2099   *         >0 otherwise. 
2100   */
2101  private long removeLastINode(final INodesInPath iip)
2102      throws QuotaExceededException {
2103    final int latestSnapshot = iip.getLatestSnapshotId();
2104    final INode last = iip.getLastINode();
2105    final INodeDirectory parent = iip.getINode(-2).asDirectory();
2106    if (!parent.removeChild(last, latestSnapshot)) {
2107      return -1;
2108    }
2109    INodeDirectory newParent = last.getParent();
2110    if (parent != newParent) {
2111      iip.setINode(-2, newParent);
2112    }
2113    
2114    if (!last.isInLatestSnapshot(latestSnapshot)) {
2115      final Quota.Counts counts = last.computeQuotaUsage();
2116      updateCountNoQuotaCheck(iip, iip.getINodes().length - 1,
2117          -counts.get(Quota.NAMESPACE), -counts.get(Quota.DISKSPACE));
2118
2119      if (INodeReference.tryRemoveReference(last) > 0) {
2120        return 0;
2121      } else {
2122        return counts.get(Quota.NAMESPACE);
2123      }
2124    }
2125    return 1;
2126  }

2684  List<XAttr> setINodeXAttr(List<XAttr> existingXAttrs, XAttr xAttr, 
2685      EnumSet<XAttrSetFlag> flag) throws QuotaExceededException, IOException {
2686    List<XAttr> xAttrs = Lists.newArrayListWithCapacity(
2687        existingXAttrs != null ? existingXAttrs.size() + 1 : 1);
2688    int userVisibleXAttrsNum = 0; // Number of user visible xAttrs
2689    boolean exist = false;
2690    if (existingXAttrs != null) {
2691      for (XAttr a: existingXAttrs) {
2692        if ((a.getNameSpace() == xAttr.getNameSpace()
2693            && a.getName().equals(xAttr.getName()))) {
2694          exist = true;
2695        } else {
2696          xAttrs.add(a);
2697          
2698          if (isUserVisible(a)) {
2699            userVisibleXAttrsNum++;
2700          }
2701        }
2702      }
2703    }
2704    
2705    XAttrSetFlag.validate(xAttr.getName(), exist, flag);
2706    xAttrs.add(xAttr);
2707    
2708    if (isUserVisible(xAttr)) {
2709      userVisibleXAttrsNum++;
2710    }
2711    
2712    if (userVisibleXAttrsNum > inodeXAttrsLimit) {
2713      throw new IOException("Cannot add additional XAttr to inode, "
2714          + "would exceed limit of " + inodeXAttrsLimit);
2715    }
2716    
2717    return xAttrs;
2718  }

2770  /**
2771   * Given an INode get all the path complents leading to it from the root.
2772   * If an Inode corresponding to C is given in /A/B/C, the returned
2773   * patch components will be {root, A, B, C}.
2774   * Note that this method cannot handle scenarios where the inode is in a
2775   * snapshot.
2776   */
2777  static byte[][] getPathComponents(INode inode) {
2778    List<byte[]> components = new ArrayList<byte[]>();
2779    components.add(0, inode.getLocalNameBytes());
2780    while(inode.getParent() != null) {
2781      components.add(0, inode.getParent().getLocalNameBytes());
2782      inode = inode.getParent();
2783    }
2784    return components.toArray(new byte[components.size()][]);
2785  }

2787  /**
2788   * The same functionality with {@link #getPathComponents(INode)}, but can
2789   * handle snapshots.
2790   */
2791  public static byte[][] getPathComponentsWithSnapshot(INode inode) {
2792    List<byte[]> components = new ArrayList<byte[]>();
2793    boolean inSnapshot = false;
2794    int snapshotId = Snapshot.CURRENT_STATE_ID;
2795    do {
2796      if (inode instanceof INodeReference.WithCount) {
2797        // identify the corresponding WithName or DstReference node
2798        inode = ((WithCount) inode).getParentRef(snapshotId);
2799      } else { // normal INode and WithName/DstReference
2800        if (inode instanceof INodeDirectory
2801            && inode.asDirectory().isSnapshottable() && inSnapshot
2802            && snapshotId != Snapshot.CURRENT_STATE_ID) {
2803          INodeDirectorySnapshottable sdir = (INodeDirectorySnapshottable) inode
2804              .asDirectory();
2805          Snapshot snapshot = sdir.getSnapshotById(snapshotId);
2806          if (snapshot != null) {
2807            components.add(0, snapshot.getRoot().getLocalNameBytes());
2808            components.add(0, HdfsConstants.DOT_SNAPSHOT_DIR_BYTES);
2809            // the snapshot has been found, thus no need to check snapshottable
2810            // directory afterwards
2811            inSnapshot = false;
2812          }
2813        }
2814        INode parent = inode.getParentReference() != null ? inode
2815            .getParentReference() : inode.getParent();
2816        if (parent != null && parent instanceof INodeDirectory) {
2817          int sid = parent.asDirectory().searchChild(inode);
2818          Preconditions.checkState(sid != Snapshot.NO_SNAPSHOT_ID);
2819          if (sid != Snapshot.CURRENT_STATE_ID
2820              && snapshotId == Snapshot.CURRENT_STATE_ID) {
2821            snapshotId = sid;
2822            inSnapshot = true;
2823          }
2824          components.add(0, inode.getLocalNameBytes());
2825        } else if (parent == null) { // root
2826          components.add(0, inode.getLocalNameBytes());
2827        }
2828        inode = parent;
2829      }
2830    } while (inode != null);
2831    return components.toArray(new byte[components.size()][]);
2832  }

2852  /**
2853   * Resolve the path of /.reserved/.inodes/<inodeid>/... to a regular path
2854   * 
2855   * @param src path that is being processed
2856   * @param pathComponents path components corresponding to the path
2857   * @param fsd FSDirectory
2858   * @return if the path indicates an inode, return path after replacing upto
2859   *         <inodeid> with the corresponding path of the inode, else the path
2860   *         in {@code src} as is.
2861   * @throws FileNotFoundException if inodeid is invalid
2862   */
2863  static String resolvePath(String src, byte[][] pathComponents, FSDirectory fsd)
2864      throws FileNotFoundException {
2865    if (pathComponents == null || pathComponents.length <= 3) {
2866      return src;
2867    }
2868    // Not /.reserved/.inodes
2869    if (!Arrays.equals(DOT_RESERVED, pathComponents[1])
2870        || !Arrays.equals(DOT_INODES, pathComponents[2])) { // Not .inodes path
2871      return src;
2872    }
2873    final String inodeId = DFSUtil.bytes2String(pathComponents[3]);
2874    final long id;
2875    try {
2876      id = Long.parseLong(inodeId);
2877    } catch (NumberFormatException e) {
2878      throw new FileNotFoundException("Invalid inode path: " + src);
2879    }
2880    if (id == INodeId.ROOT_INODE_ID && pathComponents.length == 4) {
2881      return Path.SEPARATOR;
2882    }
2883    INode inode = fsd.getInode(id);
2884    if (inode == null) {
2885      throw new FileNotFoundException(
2886          "File for given inode path does not exist: " + src);
2887    }
2888    
2889    // Handle single ".." for NFS lookup support.
2890    if ((pathComponents.length > 4)
2891        && DFSUtil.bytes2String(pathComponents[4]).equals("..")) {
2892      INode parent = inode.getParent();
2893      if (parent == null || parent.getId() == INodeId.ROOT_INODE_ID) {
2894        // inode is root, or its parent is root.
2895        return Path.SEPARATOR;
2896      } else {
2897        return parent.getFullPathName();
2898      }
2899    }
2900
2901    StringBuilder path = id == INodeId.ROOT_INODE_ID ? new StringBuilder()
2902        : new StringBuilder(inode.getFullPathName());
2903    for (int i = 4; i < pathComponents.length; i++) {
2904      path.append(Path.SEPARATOR).append(DFSUtil.bytes2String(pathComponents[i]));
2905    }
2906    if (NameNode.LOG.isDebugEnabled()) {
2907      NameNode.LOG.debug("Resolved path is " + path);
2908    }
2909    return path.toString();
2910  }
